1   package org.waarp.common.database.properties;
2   
3   import java.sql.Connection;
4   import java.sql.ResultSet;
5   import java.sql.SQLException;
6   import java.sql.Statement;
7   
8   /**
9    * MySQL Database Model
10   */
11  public class MySQLProperties extends DbProperties {
12      public static final String PROTOCOL = "mysql";
13  
14      private final String DRIVER_NAME = "com.mysql.jdbc.Driver";
15      private final String VALIDATION_QUERY = "select 1";
16      private final String MAX_CONNECTION_QUERY = "select GLOBAL_VALUE " +
17              "from INFORMATION_SCHEMA.SYSTEM_VARIABLES " +
18              "where VARIABLE_NAME LIKE 'max_connections'";
19  
20      public MySQLProperties() {
21      }
22  
23      public static String getProtocolID() {
24          return PROTOCOL;
25      }
26  
27      @Override
28      public String getDriverName() {
29          return DRIVER_NAME;
30      }
31  
32      @Override
33      public String getValidationQuery() {
34          return VALIDATION_QUERY;
35      }
36  
37      @Override
38      public int getMaximumConnections(Connection connection)
39              throws SQLException {
40          Statement stm = null;
41          ResultSet rs = null;
42          try {
43              stm = connection.createStatement();
44              rs = stm.executeQuery(MAX_CONNECTION_QUERY);
45              if (!rs.next()) {
46                  throw new SQLException("Cannot find max connection");
47              }
48              return rs.getInt(1);
49          } finally {
50              if (rs != null) {
51                  rs.close();
52              }
53              if (stm != null) {
54                  stm.close();
55              }
56          }
57      }
58  }
