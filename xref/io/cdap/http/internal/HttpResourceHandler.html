52  /**
53   * HttpResourceHandler handles the http request. HttpResourceHandler looks up all Jax-rs annotations in classes
54   * and dispatches to appropriate method on receiving requests.
55   */
56  public final class HttpResourceHandler implements HttpHandler {
57  
58    private static final Logger LOG = LoggerFactory.getLogger(HttpResourceHandler.class);
59    // Limit the number of parts of the path so that match score calculation during runtime does not overflow
60    private static final int MAX_PATH_PARTS = 25;
61  
62    private final PatternPathRouterWithGroups<HttpResourceModel> patternRouter =
63      PatternPathRouterWithGroups.create(MAX_PATH_PARTS);
64    private final Iterable<HttpHandler> handlers;
65    private final Iterable<HandlerHook> handlerHooks;
66    private final URLRewriter urlRewriter;
67  
68    /**
69     * Construct HttpResourceHandler. Reads all annotations from all the handler classes and methods passed in, constructs
70     * patternPathRouter which is routable by path to {@code HttpResourceModel} as destination of the route.
71     *
72     * @param handlers Iterable of HttpHandler.
73     * @param handlerHooks Iterable of HandlerHook.
74     * @param urlRewriter URL re-writer.
75     * @param exceptionHandler Exception handler
76     */
77    public HttpResourceHandler(Iterable<? extends HttpHandler> handlers, Iterable<? extends HandlerHook> handlerHooks,
78                               URLRewriter urlRewriter, ExceptionHandler exceptionHandler) {
79      //Store the handlers to call init and destroy on all handlers.
80      this.handlers = copyOf(handlers);
81      this.handlerHooks = copyOf(handlerHooks);
82      this.urlRewriter = urlRewriter;
83  
84      for (HttpHandler handler : handlers) {
85        LOG.trace("Parsing handler {}", handler.getClass().getName());
86        String basePath = "";
87        if (handler.getClass().isAnnotationPresent(Path.class)) {
88          basePath =  handler.getClass().getAnnotation(Path.class).value();
89        }
90  
91        for (Method method : handler.getClass().getDeclaredMethods()) {
92          Class<?>[] params = method.getParameterTypes();
93          if (params.length >= 2
94            && (params[0].isAssignableFrom(HttpRequest.class) || params[0].isAssignableFrom(FullHttpRequest.class))
95            && params[1].isAssignableFrom(HttpResponder.class)
96            && Modifier.isPublic(method.getModifiers())) {
97  
98            // For streaming consumption, the first param cannot be FullHttpMessage
99            if (BodyConsumer.class.isAssignableFrom(method.getReturnType())
100             && params[0].isAssignableFrom(FullHttpMessage.class)) {
101             throw new IllegalArgumentException(
102               "Method with return type as BodyConsumer cannot have FullHttpMessage as the first argument: " + method);
103           }
104 
105           String relativePath = "";
106           if (method.getAnnotation(Path.class) != null) {
107             relativePath = method.getAnnotation(Path.class).value();
108           }
109           String absolutePath = String.format("%s/%s", basePath, relativePath);
110           Set<HttpMethod> httpMethods = getHttpMethods(method);
111           if (httpMethods.isEmpty()) {
112             throw new IllegalArgumentException("No HttpMethod found for handler method " + method);
113           }
114           try {
115             HttpResourceModel resourceModel = new HttpResourceModel(httpMethods, absolutePath, method,
116                                                                     handler, exceptionHandler);
117             LOG.trace("Adding resource model {}", resourceModel);
118             patternRouter.add(absolutePath, resourceModel);
119           } catch (Exception e) {
120             throw new IllegalArgumentException("Failed to create http handler from method "
121                                                  + method + " in handler class " + handler.getClass().getName(), e);
122           }
123         } else {
124           LOG.trace("Not adding method {}({}) to path routing like. HTTP calls will not be routed to this method",
125                     method.getName(), params);
126         }
127       }
128     }
129   }

131   /**
132    * Fetches the HttpMethod from annotations and returns String representation of HttpMethod.
133    * Return emptyString if not present.
134    *
135    * @param method Method handling the http request.
136    * @return String representation of HttpMethod from annotations or emptyString as a default.
137    */
138   private Set<HttpMethod> getHttpMethods(Method method) {
139     Set<HttpMethod> httpMethods = new HashSet<HttpMethod>();
140 
141     if (method.isAnnotationPresent(GET.class)) {
142       httpMethods.add(HttpMethod.GET);
143     }
144     if (method.isAnnotationPresent(PUT.class)) {
145       httpMethods.add(HttpMethod.PUT);
146     }
147     if (method.isAnnotationPresent(POST.class)) {
148       httpMethods.add(HttpMethod.POST);
149     }
150     if (method.isAnnotationPresent(DELETE.class)) {
151       httpMethods.add(HttpMethod.DELETE);
152     }
153     if (method.isAnnotationPresent(OPTIONS.class)) {
154       httpMethods.add(HttpMethod.OPTIONS);
155     }
156 
157     return Collections.unmodifiableSet(httpMethods);
158   }

160   /**
161    * Call the appropriate handler for handling the httprequest. 404 if path is not found. 405 if path is found but
162    * httpMethod does not match what's configured.
163    *
164    * @param request instance of {@code HttpRequest}
165    * @param responder instance of {@code HttpResponder} to handle the request.
166    */
167   public void handle(HttpRequest request, HttpResponder responder) {
168     if (urlRewriter != null) {
169       try {
170         request.setUri(URI.create(request.uri()).normalize().toString());
171         if (!urlRewriter.rewrite(request, responder)) {
172           return;
173         }
174       } catch (Throwable t) {
175         responder.sendString(HttpResponseStatus.INTERNAL_SERVER_ERROR,
176                              String.format("Caught exception processing request. Reason: %s",
177                                           t.getMessage()));
178         LOG.error("Exception thrown during rewriting of uri {}", request.uri(), t);
179         return;
180       }
181     }
182 
183     try {
184       String path = URI.create(request.uri()).normalize().getPath();
185 
186       List<PatternPathRouterWithGroups.RoutableDestination<HttpResourceModel>> routableDestinations
187         = patternRouter.getDestinations(path);
188 
189       PatternPathRouterWithGroups.RoutableDestination<HttpResourceModel> matchedDestination =
190         getMatchedDestination(routableDestinations, request.method(), path);
191 
192       if (matchedDestination != null) {
193         //Found a httpresource route to it.
194         HttpResourceModel httpResourceModel = matchedDestination.getDestination();
195 
196         // Call preCall method of handler hooks.
197         boolean terminated = false;
198         HandlerInfo info = new HandlerInfo(httpResourceModel.getMethod().getDeclaringClass().getName(),
199                                            httpResourceModel.getMethod().getName());
200         for (HandlerHook hook : handlerHooks) {
201           if (!hook.preCall(request, responder, info)) {
202             // Terminate further request processing if preCall returns false.
203             terminated = true;
204             break;
205           }
206         }
207 
208         // Call httpresource method
209         if (!terminated) {
210           // Wrap responder to make post hook calls.
211           responder = new WrappedHttpResponder(responder, handlerHooks, request, info);
212           if (httpResourceModel.handle(request, responder, matchedDestination.getGroupNameValues()).isStreaming()) {
213             responder.sendString(HttpResponseStatus.METHOD_NOT_ALLOWED,
214                                  String.format("Body Consumer not supported for internalHttpResponder: %s",
215                                                request.uri()));
216           }
217         }
218       } else if (routableDestinations.size() > 0)  {
219         //Found a matching resource but could not find the right HttpMethod so return 405
220         responder.sendString(HttpResponseStatus.METHOD_NOT_ALLOWED,
221                              String.format("Problem accessing: %s. Reason: Method Not Allowed", request.uri()));
222       } else {
223         responder.sendString(HttpResponseStatus.NOT_FOUND, String.format("Problem accessing: %s. Reason: Not Found",
224                                                                          request.uri()));
225       }
226     } catch (Throwable t) {
227       responder.sendString(HttpResponseStatus.INTERNAL_SERVER_ERROR,
228                            String.format("Caught exception processing request. Reason: %s", t.getMessage()));
229       LOG.error("Exception thrown during request processing for uri {}", request.uri(), t);
230     }
231   }

233   /**
234    * Call the appropriate handler for handling the httprequest. 404 if path is not found. 405 if path is found but
235    * httpMethod does not match what's configured.
236    *
237    * @param request instance of {@code HttpRequest}
238    * @param responder instance of {@code HttpResponder} to handle the request.
239    * @return HttpMethodInfo object, null if urlRewriter rewrite returns false, also when method cannot be invoked.
240    */
241   @Nullable
242   public HttpMethodInfo getDestinationMethod(HttpRequest request, HttpResponder responder) throws Exception {
243     if (urlRewriter != null) {
244       try {
245         request.setUri(URI.create(request.uri()).normalize().toString());
246         if (!urlRewriter.rewrite(request, responder)) {
247           return null;
248         }
249       } catch (Throwable t) {
250         LOG.error("Exception thrown during rewriting of uri {}", request.uri(), t);
251         throw new HandlerException(HttpResponseStatus.INTERNAL_SERVER_ERROR,
252                                    String.format("Caught exception processing request. Reason: %s", t.getMessage()));
253       }
254     }
255 
256     try {
257       String path = URI.create(request.uri()).normalize().getPath();
258 
259       List<PatternPathRouterWithGroups.RoutableDestination<HttpResourceModel>> routableDestinations =
260         patternRouter.getDestinations(path);
261 
262       PatternPathRouterWithGroups.RoutableDestination<HttpResourceModel> matchedDestination =
263         getMatchedDestination(routableDestinations, request.method(), path);
264 
265       if (matchedDestination != null) {
266         HttpResourceModel httpResourceModel = matchedDestination.getDestination();
267 
268         // Call preCall method of handler hooks.
269         boolean terminated = false;
270         HandlerInfo info = new HandlerInfo(httpResourceModel.getMethod().getDeclaringClass().getName(),
271                                            httpResourceModel.getMethod().getName());
272         for (HandlerHook hook : handlerHooks) {
273           if (!hook.preCall(request, responder, info)) {
274             // Terminate further request processing if preCall returns false.
275             terminated = true;
276             break;
277           }
278         }
279 
280         // Call httpresource handle method, return the HttpMethodInfo Object.
281         if (!terminated) {
282           // Wrap responder to make post hook calls.
283           responder = new WrappedHttpResponder(responder, handlerHooks, request, info);
284           return httpResourceModel.handle(request, responder, matchedDestination.getGroupNameValues());
285         }
286       } else if (routableDestinations.size() > 0)  {
287         //Found a matching resource but could not find the right HttpMethod so return 405
288         throw new HandlerException(HttpResponseStatus.METHOD_NOT_ALLOWED, request.uri());
289       } else {
290         throw new HandlerException(HttpResponseStatus.NOT_FOUND,
291                                    String.format("Problem accessing: %s. Reason: Not Found", request.uri()));
292       }
293     } catch (Throwable t) {
294       if (t instanceof HandlerException) {
295         throw (HandlerException) t;
296       }
297       throw new HandlerException(HttpResponseStatus.INTERNAL_SERVER_ERROR,
298                                  String.format("Caught exception processing request. Reason: %s", t.getMessage()), t);
299     }
300     return null;
301   }

303   /**
304    * Get HttpResourceModel which matches the HttpMethod of the request.
305    *
306    * @param routableDestinations List of ResourceModels.
307    * @param targetHttpMethod HttpMethod.
308    * @param requestUri request URI.
309    * @return RoutableDestination that matches httpMethod that needs to be handled. null if there are no matches.
310    */
311   private PatternPathRouterWithGroups.RoutableDestination<HttpResourceModel>
312   getMatchedDestination(List<PatternPathRouterWithGroups.RoutableDestination<HttpResourceModel>> routableDestinations,
313                         HttpMethod targetHttpMethod, String requestUri) {
314 
315     LOG.trace("Routable destinations for request {}: {}", requestUri, routableDestinations);
316     Iterable<String> requestUriParts = splitAndOmitEmpty(requestUri, '/');
317     List<PatternPathRouterWithGroups.RoutableDestination<HttpResourceModel>> matchedDestinations =
318             new ArrayList<PatternPathRouterWithGroups.RoutableDestination<HttpResourceModel>>();
319     long maxScore = 0;
320 
321     for (PatternPathRouterWithGroups.RoutableDestination<HttpResourceModel> destination : routableDestinations) {
322       HttpResourceModel resourceModel = destination.getDestination();
323 
324       for (HttpMethod httpMethod : resourceModel.getHttpMethod()) {
325         if (targetHttpMethod.equals(httpMethod)) {
326           long score = getWeightedMatchScore(requestUriParts, splitAndOmitEmpty(resourceModel.getPath(), '/'));
327           LOG.trace("Max score = {}. Weighted score for {} is {}. ", maxScore, destination, score);
328           if (score > maxScore) {
329             maxScore = score;
330             matchedDestinations.clear();
331             matchedDestinations.add(destination);
332           } else if (score == maxScore) {
333             matchedDestinations.add(destination);
334           }
335         }
336       }
337     }
338 
339     if (matchedDestinations.size() > 1) {
340       throw new IllegalStateException(String.format("Multiple matched handlers found for request uri %s: %s",
341                                                     requestUri, matchedDestinations));
342     } else if (matchedDestinations.size() == 1) {
343       return matchedDestinations.get(0);
344     }
345     return null;
346   }

348   /**
349    * Generate a weighted score based on position for matches of URI parts.
350    * The matches are weighted in descending order from left to right.
351    * Exact match is weighted higher than group match, and group match is weighted higher than wildcard match.
352    *
353    * @param requestUriParts the parts of request URI
354    * @param destUriParts the parts of destination URI
355    * @return weighted score
356    */
357   private long getWeightedMatchScore(Iterable<String> requestUriParts, Iterable<String> destUriParts) {
358     // The score calculated below is a base 5 number
359     // The score will have one digit for one part of the URI
360     // This will allow for 27 parts in the path since log (Long.MAX_VALUE) to base 5 = 27.13
361     // We limit the number of parts in the path to 25 using MAX_PATH_PARTS constant above to avoid overflow during
362     // score calculation
363     long score = 0;
364     for (Iterator<String> rit = requestUriParts.iterator(), dit = destUriParts.iterator();
365          rit.hasNext() && dit.hasNext(); ) {
366       String requestPart = rit.next();
367       String destPart = dit.next();
368       if (requestPart.equals(destPart)) {
369         score = (score * 5) + 4;
370       } else if (PatternPathRouterWithGroups.GROUP_PATTERN.matcher(destPart).matches()) {
371         score = (score * 5) + 3;
372       } else {
373         score = (score * 5) + 2;
374       }
375     }
376     return score;
377   }

386   @Override

388     for (HttpHandler handler : handlers) {

390         handler.destroy(context);

392         LOG.warn("Exception raised in calling handler.destroy() for handler {}", handler, t);
393       }
394     }

397   private static <T> List<T> copyOf(Iterable<? extends T> iterable) {
398     List<T> list = new ArrayList<T>();
399     for (T item : iterable) {
400       list.add(item);
401     }
402     return Collections.unmodifiableList(list);
403   }

405   /**
406    * Helper method to split a string by a given character, with empty parts omitted.
407    */
408   private static Iterable<String> splitAndOmitEmpty(final String str, final char splitChar) {
409     return new Iterable<String>() {
410       @Override
411       public Iterator<String> iterator() {
412         return new Iterator<String>() {
413 
414           int startIdx = 0;
415           String next = null;
416 
417           @Override
418           public boolean hasNext() {
419             while (next == null && startIdx < str.length()) {
420               int idx = str.indexOf(splitChar, startIdx);
421               if (idx == startIdx) {
422                 // Omit empty string
423                 startIdx++;
424                 continue;
425               }
426               if (idx >= 0) {
427                 // Found the next part
428                 next = str.substring(startIdx, idx);
429                 startIdx = idx;
430               } else {
431                 // The last part
432                 if (startIdx < str.length()) {
433                   next = str.substring(startIdx);
434                   startIdx = str.length();
435                 }
436                 break;
437               }
438             }
439             return next != null;
440           }
441 
442           @Override
443           public String next() {
444             if (hasNext()) {
445               String next = this.next;
446               this.next = null;
447               return next;
448             }
449             throw new NoSuchElementException("No more element");
450           }
451 
452           @Override
453           public void remove() {
454             throw new UnsupportedOperationException("Remove not supported");
455           }
456         };
457       }
458     };
