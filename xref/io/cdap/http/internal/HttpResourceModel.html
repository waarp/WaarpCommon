44  /**
45   * HttpResourceModel contains information needed to handle Http call for a given path. Used as a destination in
46   * {@code PatternPathRouterWithGroups} to route URI paths to right Http end points.
47   */
48  public final class HttpResourceModel {
49  
50    private static final Set<Class<? extends Annotation>> SUPPORTED_PARAM_ANNOTATIONS =
51      Collections.unmodifiableSet(new HashSet<Class<? extends Annotation>>
52              (Arrays.asList(PathParam.class, QueryParam.class, HeaderParam.class)));
53  
54    private final Set<HttpMethod> httpMethods;
55    private final String path;
56    private final Method method;
57    private final HttpHandler handler;
58    private final List<Map<Class<? extends Annotation>, ParameterInfo<?>>> paramsInfo;
59    private final ExceptionHandler exceptionHandler;
60  
61    /**
62     * Construct a resource model with HttpMethod, method that handles httprequest, Object that contains the method.
63     *
64     * @param httpMethods Set of http methods that is handled by the resource.
65     * @param path path associated with this model.
66     * @param method handler that handles the http request.
67     * @param handler instance {@code HttpHandler}.
68     */
69    public HttpResourceModel(Set<HttpMethod> httpMethods, String path, Method method, HttpHandler handler,
70                             ExceptionHandler exceptionHandler) {
71      this.httpMethods = httpMethods;
72      this.path = path;
73      this.method = method;
74      this.handler = handler;
75      this.paramsInfo = createParametersInfos(method);
76      this.exceptionHandler = exceptionHandler;
77    }

107   /**
108    * Handle http Request.
109    *
110    * @param request  HttpRequest to be handled.
111    * @param responder HttpResponder to write the response.
112    * @param groupValues Values needed for the invocation.
113    */
114   @SuppressWarnings("unchecked")
115   public HttpMethodInfo handle(HttpRequest request,
116                                HttpResponder responder, Map<String, String> groupValues) throws Exception {
117     //TODO: Refactor group values.
118     try {
119       if (httpMethods.contains(request.method())) {
120         //Setup args for reflection call
121         Object [] args = new Object[paramsInfo.size()];
122 
123         int idx = 0;
124         for (Map<Class<? extends Annotation>, ParameterInfo<?>> info : paramsInfo) {
125           if (info.containsKey(PathParam.class)) {
126             args[idx] = getPathParamValue(info, groupValues);
127           }
128           if (info.containsKey(QueryParam.class)) {
129             args[idx] = getQueryParamValue(info, request.uri());
130           }
131           if (info.containsKey(HeaderParam.class)) {
132             args[idx] = getHeaderParamValue(info, request);
133           }
134           idx++;
135         }
136 
137         return new HttpMethodInfo(method, handler, responder, args, exceptionHandler);
138       } else {
139         //Found a matching resource but could not find the right HttpMethod so return 405
140         throw new HandlerException(HttpResponseStatus.METHOD_NOT_ALLOWED, String.format
141           ("Problem accessing: %s. Reason: Method Not Allowed", request.uri()));
142       }
143     } catch (Throwable e) {
144       throw new HandlerException(HttpResponseStatus.INTERNAL_SERVER_ERROR,
145                                  String.format("Error in executing request: %s %s", request.method(),
146                                                request.uri()), e);
147     }
148   }

150   @Override
151   public String toString() {
152     return "HttpResourceModel{" +
153       "httpMethods=" + httpMethods +
154       ", path='" + path + '\'' +
155       ", method=" + method +
156       ", handler=" + handler +
157       '}';
158   }

160   @SuppressWarnings("unchecked")
161   private Object getPathParamValue(Map<Class<? extends Annotation>, ParameterInfo<?>> annotations,
162                                    Map<String, String> groupValues) throws Exception {
163     ParameterInfo<String> info = (ParameterInfo<String>) annotations.get(PathParam.class);
164     PathParam pathParam = info.getAnnotation();
165     String value = groupValues.get(pathParam.value());
166     if (value == null) {
167       throw new IllegalArgumentException("Could not resolve value for path parameter " + pathParam.value());
168     }
169     return info.convert(value);
170   }
171 
172   @SuppressWarnings("unchecked")
173   private Object getQueryParamValue(Map<Class<? extends Annotation>, ParameterInfo<?>> annotations,
174                                     String uri) throws Exception {
175     ParameterInfo<List<String>> info = (ParameterInfo<List<String>>) annotations.get(QueryParam.class);
176     QueryParam queryParam = info.getAnnotation();
177     List<String> values = new QueryStringDecoder(uri).parameters().get(queryParam.value());
178 
179     return (values == null) ? info.convert(defaultValue(annotations)) : info.convert(values);
180   }
181 
182   @SuppressWarnings("unchecked")
183   private Object getHeaderParamValue(Map<Class<? extends Annotation>, ParameterInfo<?>> annotations,
184                                      HttpRequest request) throws Exception {
185     ParameterInfo<List<String>> info = (ParameterInfo<List<String>>) annotations.get(HeaderParam.class);
186     HeaderParam headerParam = info.getAnnotation();
187     String headerName = headerParam.value();
188     boolean hasHeader = request.headers().contains(headerName);
189     return hasHeader ? info.convert(request.headers().getAll(headerName)) : info.convert(defaultValue(annotations));
190   }

192   /**
193    * Returns a List of String created based on the {@link DefaultValue} if it is presented in the annotations Map.
194    *
195    * @return a List of String or an empty List if {@link DefaultValue} is not presented
196    */
197   private List<String> defaultValue(Map<Class<? extends Annotation>, ParameterInfo<?>> annotations) {
198     ParameterInfo<?> defaultInfo = annotations.get(DefaultValue.class);
199     if (defaultInfo == null) {
200       return Collections.emptyList();
201     }
202 
203     DefaultValue defaultValue = defaultInfo.getAnnotation();
204     return Collections.singletonList(defaultValue.value());
205   }

207   /**
208    * Gathers all parameters' annotations for the given method, starting from the third parameter.
209    */
210   private List<Map<Class<? extends Annotation>, ParameterInfo<?>>> createParametersInfos(Method method) {
211     if (method.getParameterTypes().length <= 2) {
212       return Collections.emptyList();
213     }
214 
215     List<Map<Class<? extends Annotation>, ParameterInfo<?>>> result =
216             new ArrayList<Map<Class<? extends Annotation>, ParameterInfo<?>>>();
217     Type[] parameterTypes = method.getGenericParameterTypes();
218     Annotation[][] parameterAnnotations = method.getParameterAnnotations();
219 
220     for (int i = 2; i < parameterAnnotations.length; i++) {
221       Annotation[] annotations = parameterAnnotations[i];
222       Map<Class<? extends Annotation>, ParameterInfo<?>> paramAnnotations =
223               new IdentityHashMap<Class<? extends Annotation>, ParameterInfo<?>>();
224 
225       for (Annotation annotation : annotations) {
226         Class<? extends Annotation> annotationType = annotation.annotationType();
227         ParameterInfo<?> parameterInfo;
228 
229         if (PathParam.class.isAssignableFrom(annotationType)) {
230           parameterInfo = ParameterInfo.create(annotation,
231                                                ParamConvertUtils.createPathParamConverter(parameterTypes[i]));
232         } else if (QueryParam.class.isAssignableFrom(annotationType)) {
233           parameterInfo = ParameterInfo.create(annotation,
234                                                ParamConvertUtils.createQueryParamConverter(parameterTypes[i]));
235         } else if (HeaderParam.class.isAssignableFrom(annotationType)) {
236           parameterInfo = ParameterInfo.create(annotation,
237                                                ParamConvertUtils.createHeaderParamConverter(parameterTypes[i]));
238         } else {
239           parameterInfo = ParameterInfo.create(annotation, null);
240         }
241 
242         paramAnnotations.put(annotationType, parameterInfo);
243       }
244 
245       // Must have either @PathParam, @QueryParam or @HeaderParam, but not two or more.
246       int presence = 0;
247       for (Class<? extends Annotation> annotationClass : paramAnnotations.keySet()) {
248         if (SUPPORTED_PARAM_ANNOTATIONS.contains(annotationClass)) {
249           presence++;
250         }
251       }
252       if (presence != 1) {
253         throw new IllegalArgumentException(
254           String.format("Must have exactly one annotation from %s for parameter %d in method %s",
255                         SUPPORTED_PARAM_ANNOTATIONS, i, method));
256       }
257 
258       result.add(Collections.unmodifiableMap(paramAnnotations));
259     }
260 
261     return Collections.unmodifiableList(result);
262   }

264   /**
265    * A container class to hold information about a handler method parameters.
266    */
267   private static final class ParameterInfo<T> {
268     private final Annotation annotation;
269     private final Converter<T, Object> converter;
270 
271     static <V> ParameterInfo<V> create(Annotation annotation, @Nullable Converter<V, Object> converter) {
272       return new ParameterInfo<V>(annotation, converter);
273     }
274 
275     private ParameterInfo(Annotation annotation, @Nullable Converter<T, Object> converter) {
276       this.annotation = annotation;
277       this.converter = converter;
278     }
279 
280     @SuppressWarnings("unchecked")
281     <V extends Annotation> V getAnnotation() {
282       return (V) annotation;
283     }
284 
285     Object convert(T input) throws Exception {
286       return (converter == null) ? null : converter.convert(input);
287     }
288   }
