4   import org.junit.Test;
5   import static org.junit.Assert.*;
6   
7   public class AbstractDirTest {
8       
9       @Test
10      public void testNormalizePath() {
11          assertEquals("single relative filename", "test.dat", AbstractDir.normalizePath("test.dat"));
12          assertEquals("relative path (unix)", "foo/test.dat", AbstractDir.normalizePath("foo/test.dat"));
13          assertEquals("absolute path (unix)", "/foo/test.dat", AbstractDir.normalizePath("/foo/test.dat"));
14          assertEquals("absolute uri (unix)", "file:///foo/test.dat", AbstractDir.normalizePath("file:///foo/test.dat"));
15          assertEquals("relative uri (unix)", "file://foo/test.dat", AbstractDir.normalizePath("file://foo/test.dat"));
16  
17          assertEquals("relative path (win)", "foo/test.dat", AbstractDir.normalizePath("foo\\test.dat"));
18          assertEquals("absolute path (win)", "c:/foo/test.dat", AbstractDir.normalizePath("c:\\foo\\test.dat"));
19          assertEquals("absolute uri (win)", "file:///c:/foo/test.dat", AbstractDir.normalizePath("file:///c:/foo/test.dat"));
20          assertEquals("relative uri (win)", "file://foo/test.dat", AbstractDir.normalizePath("file://foo/test.dat"));
21          assertEquals("UNC path (win)", "//server/share/test.dat", AbstractDir.normalizePath("\\\\server\\share\\test.dat"));
22          assertEquals("UNC uri (win)", "file:////server/share/test.dat", AbstractDir.normalizePath("file:////server/share/test.dat"));
23      }
24  
25      @Test
26      public void testPathFromUri() {
27          assertEquals("single relative filename", "test.dat", AbstractDir.pathFromURI("test.dat"));
28          assertEquals("relative path (unix)", "foo/test.dat", AbstractDir.pathFromURI("foo/test.dat"));
29          assertEquals("absolute path (unix)", "/foo/test.dat", AbstractDir.pathFromURI("/foo/test.dat"));
30          assertEquals("absolute uri (unix)", "/foo/test.dat", AbstractDir.pathFromURI("file:///foo/test.dat"));
31          assertEquals("relative uri (unix)", "foo/test.dat", AbstractDir.pathFromURI("file://foo/test.dat"));
32  
33          assertEquals("relative path (win)", "foo/test.dat", AbstractDir.pathFromURI("foo/test.dat"));
34          assertEquals("absolute path (win)", "c:/foo/test.dat", AbstractDir.pathFromURI("c:/foo/test.dat"));
35          assertEquals("absolute uri (win)", "c:/foo/test.dat", AbstractDir.pathFromURI("file:///c:/foo/test.dat"));
36          assertEquals("relative uri (win)", "foo/test.dat", AbstractDir.pathFromURI("file://foo/test.dat"));
37          assertEquals("UNC path (win)", "//server/share/test.dat", AbstractDir.pathFromURI("//server/share/test.dat"));
38          assertEquals("UNC uri (win)", "//server/share/test.dat", AbstractDir.pathFromURI("file:////server/share/test.dat"));
39  
40          assertEquals("url to decode", "//server/share/béah @;tré", AbstractDir.pathFromURI("file:////server/share/b%C3%A9ah%20%40%3Btr%C3%A9"));
41          assertEquals("filename with percent sign", "/tmp/100% accuracy.txt", AbstractDir.pathFromURI("file:///tmp/100% accuracy.txt"));
42      }
