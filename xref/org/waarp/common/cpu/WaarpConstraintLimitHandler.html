44      private static final String NOALERT = "noAlert";
45      public static final long LOWBANDWIDTH_DEFAULT = 1048576;
46      public String lastAlert = NOALERT;
47      private boolean constraintInactive = true;
48      private boolean useCpuLimits = false;

93      /**
94       * This constructor enables only throttling bandwidth with cpu usage
95       * 
96       * 
97       * @param WAITFORNETOP2
98       *            1000 ms as wait for a network operation
99       * @param TIMEOUTCON2
100      *            10000 ms as timeout limit
101      * @param useJdkCpuLimit
102      *            True to use JDK Cpu native or False for JavaSysMon
103      * @param lowcpuLimit
104      *            for proactive cpu limitation (throttling bandwidth) (0<= x < 1 & highcpulimit)
105      * @param highcpuLimit
106      *            for proactive cpu limitation (throttling bandwidth) (0<= x <= 1) 0 meaning no
107      *            throttle activated

109      *            for proactive cpu limitation, throttling bandwidth reduction (0 < x < 1) as 0.25
110      *            for 25% of reduction
111      * @param handler
112      *            the GlobalTrafficShapingHandler associated (null to have no proactive cpu
113      *            limitation)
114      * @param delay
115      *            the delay between 2 tests for proactive cpu limitation
116      * @param limitLowBandwidth
117      *            the minimal bandwidth (read or write) to apply when decreasing bandwidth (low
118      *            limit = 4096)
119      */

124         this(WAITFORNETOP2, TIMEOUTCON2,
125                 true, useJdkCpuLimit, 0, 0,
126                 lowcpuLimit, highcpuLimit, percentageDecrease,
127                 handler, delay, limitLowBandwidth);
128     }

148     /**
149      * This constructor enables both Connection check ability and throttling bandwidth with cpu
150      * usage
151      * 
152      * @param WAITFORNETOP2
153      *            1000 ms as wait for a network operation
154      * @param TIMEOUTCON2
155      *            10000 ms as timeout limit
156      * @param useCpuLimit
157      *            True to enable cpuLimit on connection check
158      * @param useJdKCpuLimit
159      *            True to use JDK Cpu native or False for JavaSysMon
160      * @param cpulimit
161      *            high cpu limit (0<= x < 1) to refuse new connections
162      * @param channellimit
163      *            number of connection limit (0<= x)
164      * @param lowcpuLimit
165      *            for proactive cpu limitation (throttling bandwidth) (0<= x < 1 & highcpulimit)
166      * @param highcpuLimit
167      *            for proactive cpu limitation (throttling bandwidth) (0<= x <= 1) 0 meaning no
168      *            throttle activated
169      * @param percentageDecrease
170      *            for proactive cpu limitation, throttling bandwidth reduction (0 < x < 1) as 0.25
171      *            for 25% of reduction
172      * @param handler
173      *            the GlobalTrafficShapingHandler associated (null to have no proactive cpu
174      *            limitation)
175      * @param delay
176      *            the delay between 2 tests for proactive cpu limitation
177      * @param limitLowBandwidth
178      *            the minimal bandwidth (read or write) to apply when decreasing bandwidth (low
179      *            limit = 4096)
180      */
181     public WaarpConstraintLimitHandler(long WAITFORNETOP2, long TIMEOUTCON2,
182             boolean useCpuLimit,
183             boolean useJdKCpuLimit, double cpulimit, int channellimit,
184             double lowcpuLimit, double highcpuLimit, double percentageDecrease,
185             AbstractTrafficShapingHandler handler, long delay, long limitLowBandwidth) {
186         useCpuLimits = useCpuLimit;
187         WAITFORNETOP = WAITFORNETOP2;
188         TIMEOUTCON = TIMEOUTCON2;
189         lowCpuLimit = lowcpuLimit;
190         highCpuLimit = highcpuLimit;
191         this.limitLowBandwidth = limitLowBandwidth;
192         if (this.limitLowBandwidth < LOWBANDWIDTH_DEFAULT) {
193             this.limitLowBandwidth = LOWBANDWIDTH_DEFAULT;
194         }
195         this.delay = delay;
196         if (lowCpuLimit <= 0) {
197             lowCpuLimit = highCpuLimit / 2;
198         }
199         percentageDecreaseRatio = percentageDecrease;
200         if (percentageDecreaseRatio <= 0) {
201             percentageDecreaseRatio = 0.01;
202         } else if (percentageDecreaseRatio >= 1) {
203             percentageDecreaseRatio /= 100;
204         }
205         if (delay < (WAITFORNETOP >> 1)) {
206             this.delay = WAITFORNETOP;
207         }
208         this.handler = handler;
209         if (useCpuLimits || highCpuLimit > 0) {
210             if (useJdKCpuLimit) {
211                 try {
212                     cpuManagement = new CpuManagement();
213                     constraintInactive = false;
214                 } catch (UnsupportedOperationException e) {
215                     cpuManagement = new CpuManagementNoInfo();
216                     constraintInactive = true;
217                 }
218             } else {
219                 cpuManagement = new CpuManagementSysmon();
220                 constraintInactive = false;
221             }
222         } else {
223             // no test at all
224             constraintInactive = true;
225             cpuManagement = new CpuManagementNoInfo();
226         }
227         useBandwidthLimit = highcpuLimit > 0;
228         cpuLimit = cpulimit;
229         channelLimit = channellimit;
230         lastTime = System.currentTimeMillis();
231         if (this.handler != null && (!constraintInactive) && (!useBandwidthLimit)) {
232             executor = new ScheduledThreadPoolExecutor(1);
233             executor.scheduleWithFixedDelay(this, this.delay, this.delay, TimeUnit.MILLISECONDS);
234         }
235     }

255     private double getLastLA() {
256         long newTime = System.currentTimeMillis();
257         // first check if last test was done too shortly
258         if ((newTime - lastTime) < (WAITFORNETOP >> 1)) {
259             // If last test was wrong, then redo the test
260             if (lastLA <= cpuLimit) {
261                 // last test was OK, so Continue
262                 return lastLA;
263             }
264         }
265         lastTime = newTime;
266         lastLA = cpuManagement.getLoadAverage();
267         return lastLA;
268     }
269 
270     /**
271      * 
272      * @return True if one of the limit is exceeded. Always False if not a server mode
273      */
274     public boolean checkConstraints() {
275         if (!isServer)
276             return false;
277         if ((useCpuLimits) && cpuLimit < 1 && cpuLimit > 0) {
278             getLastLA();
279             if (lastLA <= cpuLimit) {
280                 lastAlert = NOALERT;
281                 return false;
282             }
283             if (lastLA > cpuLimit) {
284                 lastAlert = "CPU Constraint: " + lastLA + " > " + cpuLimit;
285                 logger.debug(lastAlert);
286                 return true;
287             }
288         }
289         if (channelLimit > 0) {
290             int nb = DbAdmin.getNbConnection() - DbAdmin.getHttpSession();
291             if (channelLimit < nb) {
292                 lastAlert = "Network Constraint: " + nb + " > " + channelLimit;
293                 logger.debug(lastAlert);
294                 return true;
295             }
296             nb = getNumberLocalChannel();
297             if (channelLimit < nb) {
298                 lastAlert = "LocalNetwork Constraint: " + nb + " > " + channelLimit;
299                 logger.debug(lastAlert);
300                 return true;
301             }
302         }
303         lastAlert = NOALERT;
304         return false;
305     }

313     /**
314      * Same as checkConstraints except that the thread will sleep some time proportionally to the
315      * current Load (if CPU related)
316      * 
317      * @param step
318      *            the current step in retry
319      * @return True if one of the limit is exceeded. Always False if not a server mode
320      */
321     public boolean checkConstraintsSleep(int step) {
322         if (!isServer)
323             return false;
324         long delay = WAITFORNETOP >> 1;
325         if ((useCpuLimits) && cpuLimit < 1 && cpuLimit > 0) {
326             long newTime = System.currentTimeMillis();
327             // first check if last test was done too shortly
328             if ((newTime - lastTime) < delay) {
329                 // If last test was wrong, then wait a bit then redo the test
330                 if (lastLA > cpuLimit) {
331                     double sleep = lastLA * delay * (step + 1) * random.nextFloat();
332                     long shorttime = (((long) sleep) / 10) * 10;
333                     if (shorttime >= 10) {
334                         try {
335                             Thread.sleep(shorttime);
336                         } catch (InterruptedException e) {
337                         }
338                     }
339                 } else {
340                     // last test was OK, so Continue
341                     lastAlert = NOALERT;
342                     return false;
343                 }
344             }
345         }
346         if (checkConstraints()) {
347             delay = getSleepTime() * (step + 1);
348             try {
349                 Thread.sleep(delay);
350             } catch (InterruptedException e) {
351             }
352             return true;
353         } else {
354             lastAlert = NOALERT;
355             return false;
356         }
357     }

359     /**
360      * 
361      * @return a time below TIMEOUTCON with a random
362      */
363     public long getSleepTime() {
364         return (((long) (TIMEOUTCON * random.nextFloat()) + 5000) / 10) * 10;
365     }

413     /**
414      * Set the handler
415      * 
416      * @param handler
417      */
418     public void setHandler(AbstractTrafficShapingHandler handler) {
419         this.handler = handler;
420         if ((!constraintInactive) && this.handler != null && useBandwidthLimit) {
421             if (executor != null) {
422                 executor.shutdownNow();
423             }
424             logger.debug("Activate Throttle bandwidth according to CPU usage");
425             executor = new ScheduledThreadPoolExecutor(1);
426             executor.scheduleWithFixedDelay(this, this.delay, this.delay, TimeUnit.MILLISECONDS);
427         } else {
428             if (executor != null) {
429                 executor.shutdownNow();
430                 executor = null;
431             }
432         }
433     }

435     /**
436      * Check every delay if the current cpu usage needs to relax or to constraint the bandwidth
437      */
438     public void run() {
439         if (constraintInactive)
440             return;
441         double curLA = getLastLA();
442         if (!useBandwidthLimit)
443             return;
444         if (curLA > highCpuLimit) {
445             CurLimits curlimit = null;
446             if (curLimits.isEmpty()) {
447                 // get current limit setting
448                 curlimit = new CurLimits(getReadLimit(), getWriteLimit());
449                 if (curlimit.read == 0) {
450                     // take the current bandwidth
451                     curlimit.read = handler.trafficCounter().lastReadThroughput();
452                     if (curlimit.read < limitLowBandwidth) {
453                         curlimit.read = 0;
454                     }
455                 }
456                 if (curlimit.write == 0) {
457                     // take the current bandwidth
458                     curlimit.write = handler.trafficCounter().lastWriteThroughput();
459                     if (curlimit.write < limitLowBandwidth) {
460                         curlimit.write = 0;
461                     }
462                 }
463             } else {
464                 curlimit = curLimits.getLast();
465             }
466             long newread = (long) (curlimit.read * (1 - percentageDecreaseRatio));
467             if (newread < limitLowBandwidth) {
468                 newread = limitLowBandwidth;
469             }
470             long newwrite = (long) (curlimit.write * (1 - percentageDecreaseRatio));
471             if (newwrite < limitLowBandwidth) {
472                 newwrite = limitLowBandwidth;
473             }
474             CurLimits newlimit = new CurLimits(newread, newwrite);
475             if (curLimits.isEmpty() || curlimit.read != newread || curlimit.write != newwrite) {
476                 // Not same limit so add this limit
477                 curLimits.add(newlimit);
478                 logger.debug("Set new low limit since CPU = " + curLA + " " + newwrite + ":"
479                         + newread);
480                 handler.configure(newlimit.write, newlimit.read);
481                 nbSinceLastDecrease += payload;
482             }
483         } else if (curLA < lowCpuLimit) {
484             if (curLimits.isEmpty()) {
485                 // nothing to do
486                 return;
487             }
488             if (nbSinceLastDecrease > 0) {
489                 nbSinceLastDecrease--;
490                 // wait a bit more in case
491                 return;
492             }
493             nbSinceLastDecrease = 0;
494             curLimits.pollLast();
495             CurLimits newlimit = null;
496             if (curLimits.isEmpty()) {
497                 // reset to default limits
498                 long newread = getReadLimit();
499                 long newwrite = getWriteLimit();
500                 logger.debug("restore limit since CPU = " + curLA + " " + newwrite + ":" + newread);
501                 handler.configure(newwrite, newread);
502             } else {
503                 // set next upper values
504                 newlimit = curLimits.getLast();
505                 long newread = newlimit.read;
506                 long newwrite = newlimit.write;
507                 logger.debug("Set new upper limit since CPU = " + curLA + " " + newwrite + ":"
508                         + newread);
509                 handler.configure(newwrite, newread);
510                 // give extra payload to prevent a brutal return to normal
511                 nbSinceLastDecrease = payload;
512             }
513         }
514     }
