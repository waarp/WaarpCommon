333     @Test
334     public void testSetRoleDefault() {
335         for (ROLE roletoset : ROLE.values()) {
336             RoleDefault role = new RoleDefault();
337             checkRole(role, ROLE.NOACCESS);
338             role.addRole(roletoset);
339             checkRole(role, roletoset);
340             RoleDefault role2 = new RoleDefault();
341             checkRole(role2, ROLE.NOACCESS);
342             role2.setRoleDefault(role);
343             checkRole(role2, roletoset);
344             role.clear();
345             checkRole(role, ROLE.NOACCESS);
346             checkRole(role2, roletoset);
347         }
