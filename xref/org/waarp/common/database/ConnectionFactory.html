8   import java.sql.Connection;
9   import java.sql.SQLException;

11  /**
12   * A singleton wrapper of Datasource to get database connection object
13   */
14  public class ConnectionFactory {
15  
16      /**
17       * Internal Logger
18       */
19      private static final WaarpLogger logger = WaarpLoggerFactory
20          .getLogger(ConnectionFactory.class);
21  
22      /**
23       * The singleton instance
24       */
25      private static ConnectionFactory instance;
26  
27      /**
28       * DbModel
29       */
30      private final DbProperties properties;
31  
32      /**
33       * The connection url to be passed to the JDBC driver to establish a connection.
34       */
35      private final String server;
36  
37      /**
38       * The connection username to be passed to the JDBC driver to establish a connection.
39       */
40      private final String user;
41  
42      /**
43       * The connection password to be passed to the JDBC driver to establish a connection.
44       */
45      private final String password;
46  
47      private final boolean autocommit = true;
48  
49      private final boolean readonly = false;
50  
51      private int maxconnections = 50;
52  
53      /**
54       * The datasource for connection pooling
55       */
56      private BasicDataSource ds;
57  
58      /**
59       * @param server the connection url of the database
60       * @return the DbProperties Object associated with the requested URL
61       * @throws UnsupportedOperationException if the requested database
62       * is not supported
63       */
64      protected static DbProperties propertiesFor(String server)
65              throws UnsupportedOperationException {
66          if(server.contains(H2Properties.getProtocolID())) {
67              return new H2Properties();
68          } else if(server.contains(MariaDBProperties.getProtocolID())) {
69              return new MariaDBProperties();
70          } else if(server.contains(MySQLProperties.getProtocolID())) {
71              return new MySQLProperties();
72          } else if(server.contains(OracleProperties.getProtocolID())) {
73              return new OracleProperties();
74          } else if(server.contains(PostgreSQLProperties.getProtocolID())) {
75              return new PostgreSQLProperties();
76          } else {
77              throw new UnsupportedOperationException(
78                  "The requested database is not supported");
79          }
80      }
81  
82      /**
83       * Initialize the ConnectionFactory
84       * @throws UnsupportedOperationException if the requested database
85       * is not supported
86       * @throws SQLException if a error occurs while connecting to the database
87       */
88      public static void initialize(String server, String user, String password)
89              throws SQLException, UnsupportedOperationException {
90          if (instance == null) {
91               instance = new ConnectionFactory(propertiesFor(server), server,
92                   user, password);
93          }
94      }
95  
96      /**
97       * @return the initialized ConnectionFactory or
98       * null if the ConnectionFactory is not initialized
99       */
100     public static ConnectionFactory getInstance() {
101         return instance;
102     }
103 
104     /**
105      * @return a connection to the Database
106      * @throws SQLException if the ConnectionPool is not initialized
107      * or if an error occurs while accessing the database
108      */
109     public Connection getConnection() throws SQLException {
110         logger.trace("Active: {}, Idle: {}", ds.getNumActive(), ds.getNumIdle());
111         if (ds == null) {
112             throw new SQLException("ConnectionFactory is not inialized.");
113         }
114         try {
115             Connection con = ds.getConnection();
116             return con;
117         } catch (SQLException e) {
118             throw new SQLException("Cannot access database", e);
119         }
120     }
121 
122     /**
123      * @param model
124      * @param server
125      * @param user
126      * @param password
127      */
128     private ConnectionFactory(DbProperties properties, String server,
129                           String user, String password) throws SQLException {
130         this.server = server;
131         this.user = user;
132         this.password = password;
133         this.properties = properties;
134 
135         ds = new BasicDataSource();
136 
137         // Initialise DataSource
138         ds.setDriverClassName(this.properties.getDriverName());
139         ds.setUrl(this.server);
140         ds.setUsername(this.user);
141         ds.setPassword(this.password);
142         ds.setDefaultAutoCommit(autocommit);
143         ds.setDefaultReadOnly(readonly);
144         ds.setValidationQuery(this.properties.getValidationQuery());
145 
146         // Get maximum connections
147         Connection con = null;
148         try {
149             con = getConnection();
150             maxconnections = properties.getMaximumConnections(con);
151         } catch (SQLException e) {
152             logger.warn(
153                     "Cannot fetch maximum connection allowed from database", e);
154         } finally {
155             con.close();
156         }
157         ds.setMaxActive(maxconnections);
158         ds.setMaxIdle(maxconnections);
159         logger.info(this.toString());
160     }
161 
162     /**
163      * Closes and releases all registered connections and connection pool
164      */
165     public void close() {
166         logger.info("Closing ConnectionFactory");
167         try {
168             ds.close();
169         } catch (SQLException e) {
170             logger.debug("Cannot close properly the connection pool", e);
171         }
172     }
173

184     public String getUser() {

187 
188     /**
189      * @return the JDBC connection password property
190      */
191     public String getPassword() {
192         return password;
193     }
194 
195     /**
196      * @return the associated DbProperties
197      */
198     public DbProperties getProperties() {
199         return properties;
200     }
201 
202     public String toString() {
203         StringBuilder sb = new StringBuilder();
204         sb.append("Datapool:");
205         sb.append(this.server);
206         sb.append(", with user:");
207         sb.append(this.user);
208         sb.append(", AutoCommit:");
209         sb.append(this.autocommit);
210         sb.append(", DefaultReadOnly:");
211         sb.append(this.readonly);
212         sb.append(", max connecions:");
213         sb.append(this.maxconnections);
214 
215         return sb.toString();
216     }
