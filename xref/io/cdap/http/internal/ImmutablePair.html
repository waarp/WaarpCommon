17  package io.cdap.http.internal;
18  
19  import java.util.Arrays;
20  
21  /**
22   * An {@link ImmutablePair} consists of two elements within. The elements once set in the ImmutablePair cannot
23   * be modified. The class itself is final, so that it cannot be subclassed. This is general norm for creating
24   * Immutable classes. Please note that the {@link ImmutablePair} cannot be modified once set, but the objects
25   * within them can be, so in general it means that if there are mutable objects within the pair then the pair
26   * itself is effectively mutable.
27   *
28   * <pre>
29   *   ImmutablePair<Tuple, TupleInputStreamIdentifier> tupleStreamPair= new
30   *    ImmutablePair<Tuple, TupleInputStreamIdentifier> (tuple, identifier);
31   *   ...
32   *   ...
33   *   Tuple t = tupleStreamPair.getFirst();
34   *   TupleInputStreamIdentifier identifier = tupleStreamPair.getSecond();
35   *   ...
36   * </pre>
37   *
38   * @param <A> type A
39   * @param <B> type B
40   */
41  final class ImmutablePair<A, B> {
42      private final A first;
43      private final B second;
44  
45      public static <A, B> ImmutablePair<A, B> of(A first, B second) {
46          return new ImmutablePair<A, B>(first, second);
47      }
48  
49      /**
50       * Constructs a Immutable Pair.
51       * 
52       * @param first  object in pair
53       * @param second object in pair
54       */
55      private ImmutablePair(A first, B second) {
56          this.first = first;
57          this.second = second;
58      }
59  
60      /**
61       * Returns first object from pair.
62       * 
63       * @return first object from pair.
64       */
65      public A getFirst() {
66          return first;
67      }
68  
69      /**
70       * Return second object from pair.
71       * 
72       * @return second object from pair.
73       */
74      public B getSecond() {
75          return second;
76      }
77  
78      /**
79       * Returns a string representation of {@link ImmutablePair} object.
80       * 
81       * @return string representation of this object.
82       */
83      @Override
84      public String toString() {
85          return "ImmutablePair{" + "first=" + first + ", second=" + second + '}';
86      }
87  
88      /**
89       * Returns a hash code value for this object.
90       * 
91       * @return hash code value of this object.
92       */
93      @Override
94      public int hashCode() {
95          return Arrays.hashCode(new Object[] { first, second });
96      }
97  
98      /**
99       * Helper method that replaces Objects.equals(first, second)
100      * 
101      * @param first
102      * @param otherFirst
103      * @return True if equals
104      */
105     private boolean equals(Object first, Object otherFirst) {
106         boolean fst;
107         if (first == null) {
108             if (otherFirst == null) {
109                 fst = true;
110             } else {
111                 return false;
112             }
113         } else if (otherFirst == null) {
114             return false;
115         } else {
116             fst = first.equals(otherFirst);
117         }
118         return fst;
119     }
120 
121     /**
122      * Returns whether some other object "is equal" to this object.
123      * 
124      * @param o reference object with which to compare
125      * @return true if object is the same as the obj argument; false otherwise.
126      */
127     @Override
128     public boolean equals(Object o) {
129         if (o == null) {
130             return false;
131         }
132         if (!(o instanceof ImmutablePair)) {
133             return false;
134         }
135         ImmutablePair<?, ?> other = (ImmutablePair<?, ?>) o;
136         return equals(first, other.first) && equals(second, other.second);
137     }
138 }
