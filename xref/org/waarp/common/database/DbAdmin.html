137     /**
138      * Validate connection
139      * 
140      * @throws WaarpDatabaseNoConnectionException
141      */
142     public void validConnection() throws WaarpDatabaseNoConnectionException {
143         try {
144             dbModel.validConnection(getSession());
145         } catch (WaarpDatabaseNoConnectionException e) {
146             getSession().setDisActive(true);
147             setActive(false);
148             throw e;
149         }
150         getSession().setDisActive(false);
151         setActive(true);
152     }

154     /**
155      * Use a default server for basic connection. Later on, specific connection to database for the
156      * scheme that provides access to the table R66DbIndex for one specific Legacy could be done.
157      * 
158      * A this time, only one driver is possible! If a new driver is needed, then we need to create a
159      * new DbSession object. Be aware that DbSession.initialize should be call only once for each
160      * driver, whatever the number of DbSession objects that could be created (=> need a hashtable
161      * for specific driver when created). Also, don't know if two drivers at the same time (two
162      * different DbSession) is allowed by JDBC.
163      * 
164      * @param model
165      * @param server
166      * @param user
167      * @param passwd
168      * @throws WaarpDatabaseNoConnectionException
169      */
170     public DbAdmin(DbModel model, String server, String user, String passwd)
171             throws WaarpDatabaseNoConnectionException {
172         this.server = server;
173         this.user = user;
174         this.passwd = passwd;
175         this.dbModel = model;
176         this.typeDriver = model.getDbType();
177         if (typeDriver == null) {
178             logger.error("Cannot find TypeDriver");
179             throw new WaarpDatabaseNoConnectionException(
180                     "Cannot find database drive");
181         }
182         setSession(new DbSession(this, false));
183         getSession().setAdmin(this);
184         isReadOnly = false;
185         validConnection();
186         getSession().useConnection(); // default since this is the top connection
187     }

189     /**
190      * Use a default server for basic connection. Later on, specific connection to database for the
191      * scheme that provides access to the table R66DbIndex for one specific Legacy could be done.
192      * 
193      * A this time, only one driver is possible! If a new driver is needed, then we need to create a
194      * new DbSession object. Be aware that DbSession.initialize should be call only once for each
195      * driver, whatever the number of DbSession objects that could be created (=> need a hashtable
196      * for specific driver when created). Also, don't know if two drivers at the same time (two
197      * different DbSession) is allowed by JDBC.
198      * 
199      * @param model
200      * @param server
201      * @param user
202      * @param passwd
203      * @param write
204      * @throws WaarpDatabaseSqlException
205      * @throws WaarpDatabaseNoConnectionException
206      */
207     public DbAdmin(DbModel model, String server, String user, String passwd,
208             boolean write) throws WaarpDatabaseNoConnectionException {
209         this.server = server;
210         this.user = user;
211         this.passwd = passwd;
212         this.dbModel = model;
213         this.typeDriver = model.getDbType();
214         if (typeDriver == null) {
215             logger.error("Cannot find TypeDriver");
216             throw new WaarpDatabaseNoConnectionException(
217                     "Cannot find database driver");
218         }
219         if (write) {
220             for (int i = 0; i < RETRYNB; i++) {
221                 try {
222                     setSession(new DbSession(this, false));
223                 } catch (WaarpDatabaseNoConnectionException e) {
224                     logger.warn("Attempt of connection in error: " + i, e);
225                     continue;
226                 }
227                 isReadOnly = false;
228                 getSession().setAdmin(this);
229                 validConnection();
230                 getSession().useConnection(); // default since this is the top
231                                          // connection
232                 return;
233             }
234         } else {
235             for (int i = 0; i < RETRYNB; i++) {
236                 try {
237                     setSession(new DbSession(this, true));
238                 } catch (WaarpDatabaseNoConnectionException e) {
239                     logger.warn("Attempt of connection in error: " + i, e);
240                     continue;
241                 }
242                 isReadOnly = true;
243                 getSession().setAdmin(this);
244                 validConnection();
245                 getSession().useConnection(); // default since this is the top
246                                          // connection
247                 return;
248             }
249         }
250         setSession(null);
251         setActive(false);
252         logger.error("Cannot connect to Database!");
253         throw new WaarpDatabaseNoConnectionException(
254                 "Cannot connect to database");
255     }

271     /**
272      * Close the underlying session. Can be call even for connection given from the constructor
273      * DbAdmin(Connection, boolean).
274      * 
275      */
276     public void close() {
277         if (getSession() != null) {
278             getSession().endUseConnection(); // default since this is the top
279                                         // connection
280             getSession().forceDisconnect();
281             setSession(null);
282         }
283         setActive(false);
284     }

391     /**
392      * Close all database connections
393      */
394     public static void closeAllConnection() {
395         for (DbSession session : listConnection.values()) {
396             logger.debug("Close (all) Db Conn: " + session.getInternalId());
397             try {
398                 session.getConn().close();
399             } catch (SQLException e) {
400             } catch (ConcurrentModificationException e) {
401             }
402         }
403         listConnection.clear();
404         for (DbModel dbModel : DbModelFactory.dbModels) {
405             if (dbModel != null) {
406                 dbModel.releaseResources();
407             }
408         }
409         dbSessionTimer.stop();
410     }
411 
412     /**
413      * Check all database connections and try to reopen them if disActive
414      */
415     public static void checkAllConnections() {
416         for (DbSession session : listConnection.values()) {
417             try {
418                 session.checkConnection();
419             } catch (WaarpDatabaseNoConnectionException e) {
420                 logger.error("Database Connection cannot be reinitialized");
421             }
422         }
423     }

425     /**
426      * 
427      * @return True if this driver allows Thread Shared Connexion (concurrency usage)
428      */
429     public boolean isCompatibleWithThreadSharedConnexion() {
430         return (typeDriver != DbType.MariaDB && typeDriver != DbType.MySQL && typeDriver != DbType.Oracle && typeDriver != DbType.none);
