36  /**
37   * Launch the Engine from a variety of sources, either through a main() or invoked through
38   * Apache Daemon.
39   *
40   * @author Frederic Bregier
41   * Inspired from Apache Daemon Wiki
42   *
43   */
44  public abstract class ServiceLauncher implements Daemon {
45  	/**
46       * Internal Logger
47       */
48      protected static WaarpInternalLogger logger;
49  
50      protected static EngineAbstract engine = null;
51  
52      protected static ServiceLauncher engineLauncherInstance = null;
53  
54      protected ExecutorService executor = null;
55  
56      protected static DaemonController controller = null;
57      
58      protected static boolean stopCalledCorrectly = false;
59      
60      /**
61       * 
62       * @return a new EngineAbstract
63       */
64      protected abstract EngineAbstract getNewEngineAbstract();
65      
66      public ServiceLauncher() {
67      	if (logger == null) {
68      		logger = WaarpInternalLoggerFactory.getLogger(ServiceLauncher.class);
69      	}
70      	if (executor == null) {
71      		executor = Executors.newSingleThreadExecutor(new WaarpThreadFactory("ServiceLauncher"));
72      	}
73      	engineLauncherInstance = this;
74      	if (engine == null) {
75      		engine = getNewEngineAbstract();
76      	}
77      }
78      
79      protected static void initStatic() {
80      	if (!(InternalLoggerFactory.getDefaultFactory() instanceof WaarpSlf4JLoggerFactory)) {
81      		InternalLoggerFactory.setDefaultFactory(new WaarpSlf4JLoggerFactory(null));
82      	}
83      	if (logger == null) {
84      		logger = WaarpInternalLoggerFactory.getLogger(ServiceLauncher.class);
85      	}
86      	String className = Thread.currentThread().getStackTrace()[3].getClassName();;
87      	logger.debug("Engine " + className);
88      	try {
89  			engineLauncherInstance = (ServiceLauncher) Class.forName(className).newInstance();
90  		} catch (Exception e) {
91          	logger.error("Engine not correctly initialized", e);
92          	System.exit(2);
93  		}
94          if (engineLauncherInstance == null || engine == null) {
95          	logger.error("Engine not correctly initialized");
96          	System.exit(1);
97          }
98      }
99      
100     /**
101      * The Java entry point.
102      * @param args Command line arguments, all ignored.
103      */
104     public static void _main(String[] args) {
105     	initStatic();
106         // the main routine is only here so I can also run the app from the command line
107         engineLauncherInstance.initialize();
108 
109         Scanner sc = new Scanner(System.in);
110         // wait until receive stop command from keyboard
111         System.out.printf("Enter 'stop' to halt: ");
112         while(!sc.nextLine().toLowerCase().equals("stop"));
113 
114         if (!engine.isShutdown()) {
115             engineLauncherInstance.terminate();
116         }
117         sc.close();
118     }

120     /**
121      * Windows mode<br>
122      * <br>
123      * Static methods called by prunsrv to start/stop
124      * the Windows service.  Pass the argument "start"
125      * to start the service, and pass "stop" to
126      * stop the service.
127      *
128      * <pre> 
129      * prunsrv.exe //IS/MyService --Classpath=C:\...\xxx.jar --Description="My Java Service" --Jvm=auto --StartMode=jvm --StartClass=org.waarp.xxx.service.ServiceLauncher --StartMethod=windowsService --StartParams=start --StopMode=jvm --StopClass=org.waarp.xxx.service.ServiceLauncher --StopMethod=windowsService --StopParams=stop 
130      * </pre> 
131      * @param args Arguments from prunsrv command line
132      * @throws Exception 
133      **/
134     public static void _windowsService(String args[]) throws Exception {
135     	initStatic();
136         String cmd = "start";
137         if (args.length > 0) {
138             cmd = args[0];
139         }
140         if ("start".equals(cmd)) {
141             engineLauncherInstance.windowsStart();
142         } else {
143             engineLauncherInstance.windowsStop();
144         }
145     }

164     /**

166      * <br>
167      * Static methods called by prunsrv to stop
168      * the Windows service.
169      *
170      * <pre> 
171      * prunsrv.exe //IS/MyService --Classpath=C:\...\xxx.jar --Description="My Java Service" --Jvm=auto --StartMode=jvm --StartClass=org.waarp.xxx.service.ServiceLauncher --StartMethod=windowsStart --StopMode=jvm --StopClass=org.waarp.xxx.service.ServiceLauncher --StopMethod=windowsStop

181     /**
182      * Internal command
183      * @throws Exception 
184      */
185     protected void windowsStart() throws Exception {
186         logger.info("windowsStart called");
187         initialize();
188         // We must wait in Windows Mode
189         boolean status = false;
190         try {
191 			status = engine.waitShutdown();
192 		} catch (InterruptedException e) {
193 		}
194         if (!status || !stopCalledCorrectly) {
195         	// Was stopped outside service management
196         	terminate();
197         	if (controller != null) {
198         		controller.fail("Service stopped abnormally");
199         	} else {
200         		throw new Exception("Service stopped abnormally");
201         	}
202         }
203     }
204 
205     /**
206      * Internal command
207      */
208     protected void windowsStop() {
209         logger.info("windowsStop called from Service: "+stopCalledCorrectly);
210         terminate();
211         // should we force Future to be cancelled there?
212     }

218     }

221     	logger.info("Daemon start");
222         initialize();
223     }
224 
225     public void stop() {
226     	logger.info("Daemon stop");
227         terminate();
228     }

230     public void destroy() {
231     	logger.info("Daemon destroy");
232     	terminate();
233     }

235     /**
236      * Do the work of starting the engine
237      */
238     protected void initialize() {
239         if (engine != null) {
240         	logger.info("Starting the Engine");
241         	engine.setDaemon(true);
242         	executor.execute(engine);
243         } else {
244         	logger.error("Engine cannot be started since it is not initialized");
245         }
246     }
247 
248     /**
249      * Cleanly stop the engine.
250      */
251     protected void terminate() {
252         if (engine != null) {
253         	logger.info("Stopping the Engine");
254             engine.shutdown();
255             engine = null;
256         }
257     	if (executor != null) {
258     		executor.shutdown();
259     		executor = null;
260     	}
261         logger.info("Engine stopped");
262     }
