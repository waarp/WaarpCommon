18  package org.waarp.common.database.model;
19  
20  /**
21   * Type of Database supported
22   * 
23   * @author Frederic Bregier
24   * 
25   */
26  public enum DbType {
27      Oracle, MySQL, PostGreSQL, H2, MariaDB, none;
28  
29      public static DbType getFromDriver(String driver) {
30          if (driver.contains("oracle")) {
31              return DbType.Oracle;
32          } else if (driver.contains("mysql")) {
33              return DbType.MySQL;
34          } else if (driver.contains("postgresql")) {
35              return DbType.PostGreSQL;
36          } else if (driver.contains("h2")) {
37              return DbType.H2;
38          } else if (driver.contains("mariadb")) {
39              return DbType.MariaDB;
40          }
41          return none;
42      }
43  }
