50   * This package would like to propose a JSE 6 compatible way to scan a directory
51   * for new, deleted and changed files, in order to allow some functions like

87  	/**
88  	 * @param statusFile the file where the current status is saved (current files)
89  	 * @param stopFile the file when created (.exists()) will stop the daemon
90  	 * @param directory the directory where files will be monitored
91  	 * @param digest the digest to use (default if null is MD5)
92  	 * @param elapseTime the time to wait in ms for between 2 checks (default is 1000ms, minimum is 100ms)
93  	 * @param filter the filter to be applied on selected files (default is isFile())
94  	 * @param commandValidFile the commandValidFile to run (may be null, which means poll() commandValidFile has to be used)
95  	 * @param commandRemovedFile the commandRemovedFile to run (may be null)

97  	public FileMonitor(File statusFile, File stopFile,
98  			File directory, DigestAlgo digest, long elapseTime, 
99  			FileFilter filter,
100 			FileMonitorCommand commandValidFile, 
101 			FileMonitorCommand commandRemovedFile) {
102 		this.statusFile = statusFile;
103 		this.stopFile = stopFile;
104 		this.directory = directory;
105 		if (digest == null) {
106 			this.digest = defaultDigestAlgo;
107 		} else {
108 			this.digest = digest;
109 		}
110 		if (elapseTime >= minimalDelay) {
111 			this.elapseTime = (elapseTime/10)*10;
112 		}
113 		if (filter != null) {
114 			this.filter = filter;
115 		}
116 		this.commandValidFile = commandValidFile;
117 		this.commandRemovedFile = commandRemovedFile;
118 		this.reloadStatus();
119 	}

121 	protected void reloadStatus() {
122 		if (statusFile == null) return;
123 		if (! statusFile.exists()) return;
124 		try {
125 			HashMap<String, FileItem> newHashMap = 
126 					handler.mapper.readValue(statusFile, 
127 							new TypeReference<HashMap<String, FileItem>>() {});
128 			fileItems.putAll(newHashMap);
129 		} catch (JsonParseException e) {
130 		} catch (JsonMappingException e) {
131 		} catch (IOException e) {
132 		}
133 	}
134 	protected void saveStatus() {
135 		if (statusFile == null) return;
136 		try {
137 			handler.mapper.writeValue(statusFile, fileItems);
138 		} catch (JsonGenerationException e) {
139 		} catch (JsonMappingException e) {
140 		} catch (IOException e) {
141 		}
142 	}

162 	public void start() {
163 		if (timer == null) {
164 			timer = new HashedWheelTimer(
165 						new WaarpThreadFactory("TimerFileMonitor"),
166 						100, TimeUnit.MILLISECONDS, 8);
167 			timer.newTimeout(new FileMonitorTimerTask(this), elapseTime, TimeUnit.MILLISECONDS);
168 			future = new WaarpFuture(true);
169 			internalfuture = new WaarpFuture(true);
170 		}// else already started
171 	}
172 	
173 	public void stop() {
174 		if (timer != null) {
175 			timer.stop();
176 		}
177 		if (internalfuture != null) {
178 			internalfuture.awaitUninterruptibly();
179 			internalfuture = null;
180 		}
181 		timer = null;
182 		if (future != null) {
183 			future.setSuccess();
184 		}
185 	}

213 	/**
214 	 * Check Files
215 	 * @return False to stop
216 	 */
217 	protected boolean checkFiles() {
218 		boolean fileItemsChanged = false;
219 		if (stopFile.exists()) {
220 			return false;
221 		}
222 		File [] files = directory.listFiles(filter);
223 		for (File file : files) {
224 			String name = file.getName();
225 			FileItem fileItem = fileItems.get(name);
226 			if (fileItem == null) {
227 				// never seen until now
228 				fileItems.put(name, new FileItem(file));
229 				fileItemsChanged = true;
230 				continue;
231 			}
232 			if (fileItem.used) {
233 				// already used so ignore
234 				continue;
235 			}
236 			long lastTimeModified = fileItem.file.lastModified();
237 			if (lastTimeModified != fileItem.lastTime) {
238 				// changed or second time check
239 				fileItem.lastTime = lastTimeModified;
240 				fileItemsChanged = true;
241 				continue;
242 			}
243 			// now check Hash or third time
244 			try {
245 				byte [] hash = FilesystemBasedDigest.getHash(fileItem.file, true, digest);
246 				if (! Arrays.equals(hash, fileItem.hash)) {
247 					fileItem.hash = hash;
248 					fileItemsChanged = true;
249 					continue;
250 				}
251 				// now time and hash are the same so act on it
252 				fileItem.used = true;
253 				fileItem.timeUsed = System.currentTimeMillis();
254 				fileItem.hash = null;
255 				if (commandValidFile != null) {
256 					commandValidFile.run(fileItem.file);
257 				} else {
258 					toUse.add(fileItem);
259 				}
260 				fileItemsChanged = true;
261 			} catch (IOException e) {
262 				continue;
263 			}
264 		}
265 		// now check that all existing items are still valid
266 		List<FileItem> todel = new LinkedList<FileItem>();
267 		for (FileItem item : fileItems.values()) {
268 			if (item.file.isFile()) {
269 				continue;
270 			}
271 			todel.add(item);
272 		}
273 		// remove invalid files
274 		for (FileItem fileItem : todel) {
275 			fileItems.remove(fileItem.file.getName());
276 			toUse.remove(fileItem);
277 			if (commandRemovedFile != null) {
278 				commandRemovedFile.run(fileItem.file);
279 			}
280 			fileItem.file = null;
281 			fileItem.hash = null;
282 			fileItem = null;
283 			fileItemsChanged = true;
284 		}
285 		if (fileItemsChanged) {
286 			this.saveStatus();
287 		}
288 		return true;
289 	}

296 	protected static class FileMonitorTimerTask implements TimerTask {
297 		protected final FileMonitor fileMonitor;
298 		/**
299 		 * @param fileMonitor
300 		 */
301 		protected FileMonitorTimerTask(FileMonitor fileMonitor) {
302 			this.fileMonitor = fileMonitor;
303 		}
304 
305 		public void run(Timeout timeout) throws Exception {
306 			if (fileMonitor.checkFiles()) {
307 				fileMonitor.timer.newTimeout(this, fileMonitor.elapseTime, TimeUnit.MILLISECONDS);
308 			} else {
309 				fileMonitor.internalfuture.setSuccess();
310 			}
311 		}
312 		
313 	}

344 	public static void main(String[] args) {
345     	if (args.length < 3) {
346     		System.err.println("Need a statusfile, a stopfile and a directory to test");
347     		return;
348     	}
349     	File file = new File(args[0]);
350     	if (file.exists() && ! file.isFile()) {
351     		System.err.println("Not a correct status file");
352     		return;
353     	}
354     	File stopfile = new File(args[1]);
355     	if (file.exists() && ! file.isFile()) {
356     		System.err.println("Not a correct stop file");
357     		return;
358     	}
359     	File dir = new File(args[2]);
360     	if (! dir.isDirectory()) {
361     		System.err.println("Not a directory");
362     		return;
363     	}
364     	FileMonitor monitor = new FileMonitor(file, stopfile, dir, null, 0, 
365     			new RegexFileFilter(RegexFileFilter.REGEX_XML_EXTENSION), 
366     			new FileMonitorCommand() {
367 			public void run(File file) {
368 				System.out.println("File: "+file.getAbsolutePath());
369 			}
370 		}, new FileMonitorCommand() {
371 			public void run(File file) {
372 				System.err.println("File: "+file.getAbsolutePath());
373 			}
374 		});
375     	monitor.start();
376     	monitor.waitForStopFile();
377 	}
