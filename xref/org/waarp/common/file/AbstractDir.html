65       * Hack to say Windows or Unix (root like X:\ or /)

69       * Roots for Windows system

71      protected static File [] roots = null;
72      /**
73       * Init Windows Support
74       */
75      protected static void initWindowsSupport() {
76          if (ISUNIX == null) {
77              ISUNIX = (!System.getProperty("os.name")
78                      .toLowerCase().startsWith("win"));
79              if (! ISUNIX) {
80                  roots = File.listRoots();
81              }
82          }
83      }
84      /**
85      *
86      * @param file
87      * @return The corresponding Root file
88      */
89      protected File getCorrespondingRoot(File file) {
90         initWindowsSupport();
91         if (ISUNIX) {
92             return new File("/");
93         }
94         String path = file.getAbsolutePath();
95         for (File root : roots) {
96             if (path.startsWith(root.getAbsolutePath())) {
97                 return root;
98             }
99         }
100        // hack !
101        logger.warn("No root found for "+file.getAbsolutePath());
102        return roots[0];
103    }

120    public String validatePath(String path) throws CommandAbstractException {
121        String extDir;
122        if (isAbsoluteWindows(path)) {
123            extDir = path;
124            File newDir = new File(extDir);
125            return validatePath(newDir);
126        }
127        extDir = consolidatePath(path);
128        // Get the baseDir (mount point)
129        String baseDir = getSession().getAuth().getBaseDirectory();
130        // Get the translated real file path (removing '..')
131        File newDir = new File(baseDir, extDir);
132        return validatePath(newDir);
133    }
134    /**
135     *
136     * @param path
137     * @return True if the given Path is an absolute one under Windows System
138     */
139    public boolean isAbsoluteWindows(String path) {
140        initWindowsSupport();
141        if (!ISUNIX) {
142            File file = new File(path);
143            return file.isAbsolute();
144        }
145        return false;
146    }
147    /**
148     * Consolidate Path as relative or absolute path to an absolute path
149     *
150     * @param path
151     * @return the consolidated path
152     * @throws CommandAbstractException
153     */
154    protected String consolidatePath(String path)
155            throws CommandAbstractException {
156        if (path == null || path.length() == 0) {
157            throw new Reply501Exception("Path must not be empty");
158        }
159        // First check if the path is relative or absolute
160        if (isAbsoluteWindows(path)) {
161            return path;
162        }
163        String extDir = null;
164        if (path.charAt(0) == SEPARATORCHAR) {
165            extDir = path;
166        } else {
167            extDir = currentDir + SEPARATOR + path;
168        }
169        return extDir;
170    }
171    /**
172     * Construct the CanonicalPath without taking into account symbolic link
173     * @param dir
174     * @return the canonicalPath
175     */
176    protected String getCanonicalPath(File dir) {
177        initWindowsSupport();
178        if (ISUNIX) {
179            // resolve it without getting symbolic links
180            StringBuilder builder = new StringBuilder();
181            // Get the path in reverse order from end to start
182            List<String> list = new ArrayList<String>();
183            File newdir = dir;
184            String lastdir = newdir.getName();
185            list.add(lastdir);
186            File parent = newdir.getParentFile();
187            while (parent != null) {
188                newdir = parent;
189                lastdir = newdir.getName();
190                list.add(lastdir);
191                parent = newdir.getParentFile();
192            }
193            // Now filter on '..' or '.'
194            for (int i = list.size()-1; i >= 0; i--) {
195                String curdir = list.get(i);
196                if (curdir.equals(".")) {
197                    list.remove(i);// removes '.'
198                } else if (curdir.equals("..")) {
199                        list.remove(i);// removes '..'
200                        int len = list.size();
201                        if (len > 0 && i < len) {
202                            list.remove(i);// and removes parent dir
203                        }
204                }
205            }
206            if (list.isEmpty()) {
207                return "/";
208            }
209 
210            for (int i = list.size()-1; i >= 0; i--) {
211                builder.append('/');
212                builder.append(list.get(i));
213            }
214            return builder.toString();
215        }
216        //Windows version
217        // no link so just use the default version of canonical Path
218        try {
219            return dir.getCanonicalPath();
220        } catch (IOException e) {
221            return dir.getAbsolutePath();
222        }
223    }
224    /**
225     * Same as validatePath but from a FileInterface
226     *
227     * @param dir
228     * @return the construct and validated path (could be different than the one
229     *         given as argument, example: '..' are removed)
230     * @throws CommandAbstractException
231     */
232    protected String validatePath(File dir) throws CommandAbstractException {
233        String extDir = null;
234        extDir = normalizePath(getCanonicalPath(dir));
235        // Get the relative business path
236        extDir = getSession().getAuth().getRelativePath(extDir);
237        // Check if this business path is valid
238        if (getSession().getAuth().isBusinessPathValid(extDir)) {
239            return extDir;
240        }
241        throw new Reply553Exception("Pathname not allowed");
242    }

265    public FileInterface setFile(String path,
266            boolean append) throws CommandAbstractException {
267        checkIdentify();
268        String newpath = consolidatePath(path);
269        List<String> paths = wildcardFiles(newpath);
270        if (paths.size() != 1) {
271            throw new Reply550Exception("File not found: " +
272                    paths.size() + " founds");
273        }
274        String extDir = paths.get(0);
275        return newFile(extDir, append);
276    }
