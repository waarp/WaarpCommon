18  package org.waarp.common.command;
19  
20  import org.waarp.common.command.exception.CommandAbstractException;
21  import org.waarp.common.exception.InvalidArgumentException;
22  import org.waarp.common.file.SessionInterface;
23  
24  /**
25   * Interface for Command
26   * 
27   * @author Frederic Bregier
28   * 
29   */
30  public interface CommandInterface {
31      /**
32       * Set the Command from the args
33       * 
34       * @param session
35       * @param command
36       * @param arg
37       * @param code
38       */
39      public void setArgs(SessionInterface session, String command, String arg,
40              @SuppressWarnings("rawtypes") Enum code);
41  
42      /**
43       * Execute the command. This execution must set the replyCode in the session to a correct value
44       * before returning.
45       * 
46       * @exception CommandAbstractException
47       *                in case of an FTP Error occurs
48       */
49      abstract public void exec() throws CommandAbstractException;
50  
51      /**
52       * This function is intend to allow to force USER->PASS->ACCT->CDW for instance
53       * 
54       * @param extraNextCommand
55       *            the extraNextCommand to set
56       */
57      public void setExtraNextCommand(@SuppressWarnings("rawtypes") Enum extraNextCommand);
58  
59      /**
60       * This function is called when a new command is received to check if this new command is
61       * positive according to the previous command and status.
62       * 
63       * @param newCommand
64       * @return True if this new command is OK, else False
65       */
66      public abstract boolean isNextCommandValid(CommandInterface newCommand);
67  
68      /**
69       * @return the object
70       */
71      public Object getObject();
72  
73      /**
74       * @param object
75       *            the object to set
76       */
77      public void setObject(Object object);
78  
79      /**
80       * @return the arg
81       */
82      public String getArg();
83  
84      /**
85       * 
86       * @return the list of arguments
87       */
88      public String[] getArgs();
89  
90      /**
91       * Get an integer value from argument
92       * 
93       * @param argx
94       * @return the integer
95       * @throws InvalidArgumentException
96       *             if the argument is not an integer
97       */
98      public int getValue(String argx) throws InvalidArgumentException;
99  
100     /**
101      * @return the command
102      */
103     public String getCommand();
104 
105     /**
106      * Does this command has an argument
107      * 
108      * @return True if it has an argument
109      */
110     public boolean hasArg();
111 
112     /**
113      * 
114      * @return the current SessionInterface
115      */
116     public SessionInterface getSession();
117 
118     // some helpful functions
119     /**
120      * Set the previous command as the new current command (used after a incorrect sequence of
121      * commands or unknown command). Also clear the Restart object.
122      * 
123      */
124     public void invalidCurrentCommand();
125 
126     /**
127      * 
128      * @return The GgCommandCode associated with this command
129      */
130     public Enum<?> getCode();
131 }
