18  package org.waarp.common.utility;
19  
20  import io.netty.bootstrap.Bootstrap;
21  import io.netty.bootstrap.ServerBootstrap;
22  import io.netty.buffer.PooledByteBufAllocator;
23  import io.netty.channel.ChannelOption;
24  import io.netty.channel.EventLoopGroup;
25  import io.netty.channel.socket.nio.NioServerSocketChannel;
26  import io.netty.channel.socket.nio.NioSocketChannel;
27  
28  /**
29   * Utility class for Netty usage
30   * 
31   * @author "Frederic Bregier"
32   *
33   */
34  public class WaarpNettyUtil {
35  
36      private WaarpNettyUtil() {
37      }
38  
39      /**
40       * Add default configuration for client bootstrap
41       * 
42       * @param bootstrap
43       * @param group
44       * @param timeout
45       */
46      public static void setBootstrap(Bootstrap bootstrap, EventLoopGroup group, int timeout) {
47          bootstrap.channel(NioSocketChannel.class);
48          bootstrap.group(group);
49          bootstrap.option(ChannelOption.TCP_NODELAY, true);
50          bootstrap.option(ChannelOption.SO_REUSEADDR, true);
51          bootstrap.option(ChannelOption.SO_KEEPALIVE, true);
52          bootstrap.option(ChannelOption.CONNECT_TIMEOUT_MILLIS, timeout);
53          bootstrap.option(ChannelOption.SO_RCVBUF, 1048576);
54          bootstrap.option(ChannelOption.SO_SNDBUF, 1048576);
55          bootstrap.option(ChannelOption.WRITE_BUFFER_HIGH_WATER_MARK, 10*65536);
56          bootstrap.option(ChannelOption.WRITE_BUFFER_LOW_WATER_MARK, 2*65536);
57          bootstrap.option(ChannelOption.ALLOCATOR, PooledByteBufAllocator.DEFAULT);
58      }
59  
60      /**
61       * Add default configuration for server bootstrap
62       * 
63       * @param bootstrap
64       * @param groupBoss
65       * @param groupWorker
66       * @param timeout
67       */
68      public static void setServerBootstrap(ServerBootstrap bootstrap, EventLoopGroup groupBoss,
69              EventLoopGroup groupWorker, int timeout) {
70          bootstrap.channel(NioServerSocketChannel.class);
71          bootstrap.group(groupBoss, groupWorker);
72          bootstrap.option(ChannelOption.TCP_NODELAY, true);
73          bootstrap.option(ChannelOption.SO_REUSEADDR, true);
74          bootstrap.childOption(ChannelOption.TCP_NODELAY, true);
75          bootstrap.childOption(ChannelOption.SO_REUSEADDR, true);
76          bootstrap.childOption(ChannelOption.SO_KEEPALIVE, true);
77          bootstrap.childOption(ChannelOption.CONNECT_TIMEOUT_MILLIS, timeout);
78          bootstrap.childOption(ChannelOption.SO_RCVBUF, 1048576);
79          bootstrap.childOption(ChannelOption.SO_SNDBUF, 1048576);
80          bootstrap.childOption(ChannelOption.WRITE_BUFFER_HIGH_WATER_MARK, 10*65536);
81          bootstrap.childOption(ChannelOption.WRITE_BUFFER_LOW_WATER_MARK, 2*65536);
82          bootstrap.childOption(ChannelOption.ALLOCATOR, PooledByteBufAllocator.DEFAULT);
83      }
84  }
