63      /**
64       * Initialize empty KeyStore. No TrustStore is internally created.
65       * 
66       * @param _keyStorePasswd
67       * @param _keyPassword
68       * @throws CryptoException
69       */
70      public WaarpSecureKeyStore(String _keyStorePasswd, String _keyPassword) throws CryptoException {
71          keyStorePasswd = _keyStorePasswd;
72          keyPassword = _keyPassword;
73          try {
74              keyStore = KeyStore.getInstance("JKS");
75          } catch (KeyStoreException e) {
76              logger.error("Cannot create KeyStore Instance", e);
77              throw new CryptoException("Cannot create KeyStore Instance", e);
78          }
79          try {
80              // Empty keyStore created so null for the InputStream
81              keyStore.load(null,
82                      getKeyStorePassword());
83          } catch (NoSuchAlgorithmException e) {
84              logger.error("Cannot create KeyStore Instance", e);
85              throw new CryptoException("Cannot create KeyStore Instance", e);
86          } catch (CertificateException e) {
87              logger.error("Cannot create KeyStore Instance", e);
88              throw new CryptoException("Cannot create KeyStore Instance", e);
89          } catch (FileNotFoundException e) {
90              logger.error("Cannot create KeyStore Instance", e);
91              throw new CryptoException("Cannot create KeyStore Instance", e);
92          } catch (IOException e) {
93              logger.error("Cannot create KeyStore Instance", e);
94              throw new CryptoException("Cannot create KeyStore Instance", e);
95          }
96          initKeyManagerFactory();
97      }

113     /**
114      * Initialize the SecureKeyStore and TrustStore from files
115      * 
116      * @param keyStoreFilename
117      * @param _keyStorePasswd
118      * @param _keyPassword
119      * @param trustStoreFilename
120      *            if Null, no TrustKeyStore will be created
121      * @param _trustStorePasswd
122      * @param needClientAuthent
123      *            True if the TrustStore is also used for Client Authentication
124      * @throws CryptoException
125      */
126     public WaarpSecureKeyStore(
127             String keyStoreFilename, String _keyStorePasswd, String _keyPassword,
128             String trustStoreFilename, String _trustStorePasswd, boolean needClientAuthent)
129             throws CryptoException {
130         // Create the KeyStore
131         initKeyStore(keyStoreFilename, _keyStorePasswd, _keyPassword);
132         // Now create the TrustKeyStore
133         if (trustStoreFilename != null) {
134             initTrustStore(trustStoreFilename, _trustStorePasswd, needClientAuthent);
135         } else {
136             initEmptyTrustStore();
137         }
138     }

140     /**
141      * Initialize the SecureKeyStore with no TrustStore from file
142      * 
143      * @param _keyStoreFilename
144      * @param _keyStorePasswd
145      * @param _keyPassword
146      * @throws CryptoException
147      */
148     public void initKeyStore(String _keyStoreFilename, String _keyStorePasswd, String _keyPassword)
149             throws CryptoException {
150         keyStoreFilename = _keyStoreFilename;
151         keyStorePasswd = _keyStorePasswd;
152         keyPassword = _keyPassword;
153         // First keyStore itself
154         try {
155             keyStore = KeyStore.getInstance("JKS");
156         } catch (KeyStoreException e) {
157             logger.error("Cannot create KeyStore Instance", e);
158             throw new CryptoException("Cannot create KeyStore Instance", e);
159         }
160         FileInputStream inputStream = null;
161         try {
162             inputStream = new FileInputStream(keyStoreFilename);
163             keyStore.load(inputStream, getKeyStorePassword());
164         } catch (NoSuchAlgorithmException e) {
165             logger.error("Cannot create KeyStore Instance", e);
166             throw new CryptoException("Cannot create KeyStore Instance", e);
167         } catch (CertificateException e) {
168             logger.error("Cannot create KeyStore Instance", e);
169             throw new CryptoException("Cannot create KeyStore Instance", e);
170         } catch (FileNotFoundException e) {
171             logger.error("Cannot create KeyStore Instance", e);
172             throw new CryptoException("Cannot create KeyStore Instance", e);
173         } catch (IOException e) {
174             logger.error("Cannot create KeyStore Instance", e);
175             throw new CryptoException("Cannot create KeyStore Instance", e);
176         } finally {
177             try {
178                 if (inputStream != null) {
179                     inputStream.close();
180                 }
181             } catch (IOException e) {
182             }
183         }
184         initKeyManagerFactory();
185     }

187     /**
188      * Init KeyManagerFactory
189      * 
190      * @throws CryptoException
191      */
192     void initKeyManagerFactory() throws CryptoException {
193         try {
194             keyManagerFactory = KeyManagerFactory.getInstance(
195                     KeyManagerFactory.getDefaultAlgorithm());
196             // "SunX509");
197         } catch (NoSuchAlgorithmException e) {
198             logger.error("Cannot create KeyManagerFactory Instance", e);
199             throw new CryptoException("Cannot create KeyManagerFactory Instance", e);
200         }
201         try {
202             keyManagerFactory.init(keyStore, getCertificatePassword());
203         } catch (UnrecoverableKeyException e) {
204             logger.error("Cannot create KeyManagerFactory Instance", e);
205             throw new CryptoException("Cannot create KeyManagerFactory Instance", e);
206         } catch (KeyStoreException e) {
207             logger.error("Cannot create KeyManagerFactory Instance", e);
208             throw new CryptoException("Cannot create KeyManagerFactory Instance", e);
209         } catch (NoSuchAlgorithmException e) {
210             logger.error("Cannot create KeyManagerFactory Instance", e);
211             throw new CryptoException("Cannot create KeyManagerFactory Instance", e);
212         }
213     }

216      * Delete a Key from the KeyStore based on its alias
217      * 
218      * @param alias
219      * @return True if entry is deleted
220      */
221     public boolean deleteKeyFromKeyStore(String alias) {
222         try {
223             keyStore.deleteEntry(alias);
224         } catch (KeyStoreException e) {
225             logger.error("Cannot delete Key from KeyStore Instance", e);
226             return false;
227         }
228         return true;

249     /**
250      * Save a KeyStore to a file
251      * 
252      * @param filename
253      * @return True if keyStore is saved to file
254      */
255     public boolean saveKeyStore(String filename) {
256         FileOutputStream fos = null;
257         try {
258             fos = new FileOutputStream(filename);
259             try {
260                 keyStore.store(fos, getKeyStorePassword());
261             } catch (KeyStoreException e) {
262                 logger.error("Cannot save to file KeyStore Instance", e);
263                 return false;
264             } catch (NoSuchAlgorithmException e) {
265                 logger.error("Cannot save to file KeyStore Instance", e);
266                 return false;
267             } catch (CertificateException e) {
268                 logger.error("Cannot save to file KeyStore Instance", e);
269                 return false;
270             } catch (IOException e) {
271                 logger.error("Cannot save to file KeyStore Instance", e);
272                 return false;
273             }
274         } catch (FileNotFoundException e) {
275             logger.error("Cannot save to file KeyStore Instance", e);
276             return false;
277         } finally {
278             try {
279                 if (fos != null) {
280                     fos.close();
281                 }
282             } catch (IOException e) {
283             }
284         }
285         return true;
286     }

288     /**
289      * Initialize the TrustStore from a filename and its password
290      * 
291      * @param _trustStoreFilename
292      * @param _trustStorePasswd
293      * @param needClientAuthent
294      *            True if the TrustStore is also to authenticate clients
295      * @throws CryptoException
296      */
297     public void initTrustStore(String _trustStoreFilename, String _trustStorePasswd,
298             boolean needClientAuthent) throws CryptoException {
299         trustStoreFilename = _trustStoreFilename;
300         trustStorePasswd = _trustStorePasswd;
301         try {
302             keyTrustStore = KeyStore.getInstance("JKS");
303         } catch (KeyStoreException e) {
304             logger.error("Cannot create TrustManagerFactory Instance", e);
305             throw new CryptoException("Cannot create TrustManagerFactory Instance", e);
306         }
307         FileInputStream inputStream = null;
308         try {
309             inputStream = new FileInputStream(trustStoreFilename);
310             keyTrustStore.load(inputStream, getKeyTrustStorePassword());
311         } catch (NoSuchAlgorithmException e) {
312             logger.error("Cannot create TrustManagerFactory Instance", e);
313             throw new CryptoException("Cannot create TrustManagerFactory Instance", e);
314         } catch (CertificateException e) {
315             logger.error("Cannot create TrustManagerFactory Instance", e);
316             throw new CryptoException("Cannot create TrustManagerFactory Instance", e);
317         } catch (FileNotFoundException e) {
318             logger.error("Cannot create TrustManagerFactory Instance", e);
319             throw new CryptoException("Cannot create TrustManagerFactory Instance", e);
320         } catch (IOException e) {
321             logger.error("Cannot create TrustManagerFactory Instance", e);
322             throw new CryptoException("Cannot create TrustManagerFactory Instance", e);
323         } finally {
324             try {
325                 if (inputStream != null) {
326                     inputStream.close();
327                 }
328             } catch (IOException e2) {
329             }
330         }
331         TrustManagerFactory trustManagerFactory = null;
332         try {
333             trustManagerFactory = TrustManagerFactory.getInstance(
334                     KeyManagerFactory.getDefaultAlgorithm());
335         } catch (NoSuchAlgorithmException e1) {
336             logger.error("Cannot create TrustManagerFactory Instance", e1);
337             throw new CryptoException("Cannot create TrustManagerFactory Instance", e1);
338         }
339         try {
340             trustManagerFactory.init(keyTrustStore);
341         } catch (KeyStoreException e1) {
342             logger.error("Cannot create TrustManagerFactory Instance", e1);
343             throw new CryptoException("Cannot create TrustManagerFactory Instance", e1);
344         }
345         try {
346             secureTrustManagerFactory = new WaarpSecureTrustManagerFactory(trustManagerFactory,
347                     needClientAuthent);
348         } catch (CryptoException e) {
349             logger.error("Cannot create TrustManagerFactory Instance", e);
350             throw new CryptoException("Cannot create TrustManagerFactory Instance", e);
351         }
352     }

354     /**
355      * Initialize an empty TrustStore
356      * 
357      * @return True if correctly initialized empty
358      */
359     public boolean initEmptyTrustStore() {
360         trustStorePasswd = "secret";
361         try {
362             keyTrustStore = KeyStore.getInstance("JKS");
363         } catch (KeyStoreException e) {
364             logger.error("Cannot create keyTrustStore Instance", e);
365             return false;
366         }
367         try {
368             // Empty keyTrustStore created so null for the InputStream
369             keyTrustStore.load(null,
370                     getKeyTrustStorePassword());
371         } catch (NoSuchAlgorithmException e) {
372             logger.error("Cannot create keyTrustStore Instance", e);
373             return false;
374         } catch (CertificateException e) {
375             logger.error("Cannot create keyTrustStore Instance", e);
376             return false;
377         } catch (FileNotFoundException e) {
378             logger.error("Cannot create keyTrustStore Instance", e);
379             return false;
380         } catch (IOException e) {
381             logger.error("Cannot create keyTrustStore Instance", e);
382             return false;
383         }
384         secureTrustManagerFactory = new WaarpSecureTrustManagerFactory();
385         return true;
386     }

460     /**
461      * Load a certificate from a filename
462      * 
463      * @param filename
464      * @return the X509 Certificate from filename
465      * @throws CertificateException
466      * @throws FileNotFoundException
467      */
468     public static Certificate loadX509Certificate(String filename)
469             throws CertificateException, FileNotFoundException {
470         CertificateFactory cf = CertificateFactory.getInstance("X.509");
471         FileInputStream in = new FileInputStream(filename);
472         try {
473             return cf.generateCertificate(in);
474         } finally {
475             try {
476                 in.close();
477             } catch (IOException e) {
478             }
479         }
480     }
