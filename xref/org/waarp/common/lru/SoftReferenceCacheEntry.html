18  package org.waarp.common.lru;
19  
20  import java.lang.ref.SoftReference;
21  
22  /**
23   * Cache entry which uses SoftReference to store value
24   * 
25   * @author Frederic Bregier
26   * @author Damian Momot
27   */
28  class SoftReferenceCacheEntry<V> implements InterfaceLruCacheEntry<V> {
29  
30  	private final SoftReference<V> valueReference;
31  
32  	private long expirationTime;
33  
34  	/**
35  	 * Creates LruCacheEntry with desired ttl
36  	 * 
37  	 * @param value
38  	 * @param ttl
39  	 *            time to live in milliseconds
40  	 * @throws IllegalArgumentException
41  	 *             if ttl is not positive
42  	 */
43  	SoftReferenceCacheEntry(V value, long ttl) {
44  		if (ttl <= 0)
45  			throw new IllegalArgumentException("ttl must be positive");
46  
47  		valueReference = new SoftReference<V>(value);
48  		expirationTime = System.currentTimeMillis() + ttl;
49  	}
50  
51  	/**
52  	 * Returns value if entry is valid, null otherwise.
53  	 * 
54  	 * Entry is invalid if SoftReference is cleared or entry has expired
55  	 * 
56  	 * @return value if entry is valid
57  	 */
58  	public V getValue() {
59  		V value = null;
60  
61  		// check expiration time
62  		if (System.currentTimeMillis() <= expirationTime) {
63  			value = valueReference.get();
64  
65  			// if (value == null)
66  			// logger.warn("SoftReferency.get() returned null - probably JVM runs out of memory");
67  		}
68  
69  		return value;
70  	}
71  
72  	public boolean isStillValid(long timeRef) {
73  		return (timeRef <= expirationTime);
74  	}
75  
76  	public boolean resetTime(long ttl) {
77  		expirationTime = System.currentTimeMillis() + ttl;
78  		return true;
79  	}
80  }
