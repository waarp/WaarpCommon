42      /**
43       * Test method for {@link org.waarp.common.json.JsonHandler#createObjectNode()}.
44       */
45      @Test
46      public void testCreateObjectNode() {
47          ObjectNode node = JsonHandler.createObjectNode();
48          JsonHandler.setValue(node, "boolean", true);
49          JsonHandler.setValue(node, "bytes", "bytes".getBytes());
50          JsonHandler.setValue(node, "double", 2.0);
51          JsonHandler.setValue(node, "int", 3);
52          JsonHandler.setValue(node, "long", 5L);
53          JsonHandler.setValue(node, "string", "string");
54          assertTrue(JsonHandler.exist(node, "boolean"));
55          assertTrue(JsonHandler.exist(node, "bytes"));
56          assertTrue(JsonHandler.exist(node, "double"));
57          assertTrue(JsonHandler.exist(node, "int"));
58          assertTrue(JsonHandler.exist(node, "long"));
59          assertTrue(JsonHandler.exist(node, "string"));
60          assertEquals(true, JsonHandler.getValue(node, "boolean", false));
61          assertTrue(Arrays.equals("bytes".getBytes(), JsonHandler.getValue(node, "bytes", " ".getBytes())));
62          assertEquals(2.0, (double) JsonHandler.getValue(node, "double", 1.0), 0.0);
63          assertEquals(3, (int) JsonHandler.getValue(node, "int", 2));
64          assertEquals(5L, (long) JsonHandler.getValue(node, "long", 0L));
65          assertEquals("string", JsonHandler.getValue(node, "string", "no"));
66          String result = JsonHandler.writeAsString(node);
67          ObjectNode node2 = JsonHandler.getFromString(result);
68          assertTrue(result.equals(JsonHandler.writeAsString(node2)));
69          Map<String, Object> map = JsonHandler.getMapFromString(result);
70          assertEquals(true, map.get("boolean"));
71          assertEquals(2.0, (Double) map.get("double"), 0.0);
72          assertTrue(3 == (Integer) map.get("int"));
73          assertTrue(5L == (Integer) map.get("long"));
74          assertEquals("string", map.get("string"));
75      }

77      /**
78       * Test method for {@link org.waarp.common.json.JsonHandler#createArrayNode()}.
79       */
80      @Test
81      public void testCreateArrayNode() {
82          ArrayNode node = JsonHandler.createArrayNode();
83          node.add(true);
84          node.add("bytes".getBytes());
85          node.add(2.0);
86          node.add(3);
87          node.add(5L);
88          node.add("string");
89          assertTrue(node.size() == 6);
90          String result = JsonHandler.writeAsString(node);
91          ArrayNode node2;
92          try {
93              node2 = (ArrayNode) JsonHandler.mapper.readTree(result);
94          } catch (JsonProcessingException e) {
95              fail(e.getMessage());
96              return;
97          } catch (IOException e) {
98              fail(e.getMessage());
99              return;
100         }
101         assertTrue(result.equals(JsonHandler.writeAsString(node2)));
102     }
