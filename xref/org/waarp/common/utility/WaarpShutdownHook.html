88      public WaarpShutdownHook(ShutdownConfiguration configuration) {
89          if (initialized) {
90              shutdownHook.shutdownConfiguration = configuration;
91              this.setName("WaarpShutdownHook");
92              this.setDaemon(true);
93              shutdownHook = this;
94              this.shutdownConfiguration = configuration;
95              return;
96          }
97          this.shutdownConfiguration = configuration;
98          this.setName("WaarpShutdownHook");
99          this.setDaemon(true);
100         shutdownHook = this;
101         initialized = true;
102     }

146     /**
147      * This function is the top function to be called when the process is to be shutdown.
148      * 
149      * @param immediateSet
150      */
151     public static void terminate(boolean immediateSet) {
152         if (immediateSet) {
153             immediate = immediateSet;
154         }
155         if (shutdownHook != null) {
156             removeShutdownHook();
157             terminate();
158         } else {
159             logger.error("No ShutdownHook setup");
160             System.exit(1);
161         }
162     }
163 
164     @Override
165     public void run() {
166         if (isShutdownOver) {
167             if (shutdownHook != null && shutdownHook.serviceStopped()) {
168                 try {
169                     Thread.sleep(100);
170                 } catch (InterruptedException e) {
171                 }
172             }
173             // Already stopped
174             System.err.println("Halt System now - services already stopped -");
175             Runtime.getRuntime().halt(0);
176             return;
177         }
178         try {
179             terminate();
180         } catch (Throwable t) {
181             if (shutdownHook != null && shutdownHook.serviceStopped()) {
182                 try {
183                     Thread.sleep(100);
184                 } catch (InterruptedException e) {
185                 }
186             }
187         }
188         System.err.println("Halt System now");
189         Runtime.getRuntime().halt(0);
190     }

192     /**
193      * Print stack trace
194      * 
195      * @param thread
196      * @param stacks
197      */
198     static private void printStackTrace(Thread thread, StackTraceElement[] stacks) {
199         System.err.print(thread.toString() + " : ");
200         for (int i = 0; i < stacks.length - 1; i++) {
201             System.err.print(stacks[i].toString() + " ");
202         }
203         if (stacks.length >= 1)
204             System.err.println(stacks[stacks.length - 1].toString());
205         else
206             System.err.println();
207     }

209     /**
210      * Finalize resources attached to handlers
211      * 
212      * @author Frederic Bregier
213      */
214     private static class ShutdownTimerTask extends TimerTask {
215         /**
216          * Internal Logger
217          */
218         private static final WaarpLogger logger = WaarpLoggerFactory
219                 .getLogger(ShutdownTimerTask.class);
220 
221         /**
222          * Constructor from type
223          * 
224          * @param type
225          */
226         private ShutdownTimerTask() {
227         }
228 
229         @Override
230         public void run() {
231             System.err.println("Halt System now - time waiting is over");
232             logger.error("System will force EXIT");
233             if (shutdownHook != null && shutdownHook.serviceStopped()) {
234                 try {
235                     Thread.sleep(100);
236                 } catch (InterruptedException e) {
237                 }
238             }
239             if (logger.isDebugEnabled()) {
240                 Map<Thread, StackTraceElement[]> map = Thread
241                         .getAllStackTraces();
242                 for (Thread thread : map.keySet()) {
243                     printStackTrace(thread, map.get(thread));
244                 }
245             }
246             Runtime.getRuntime().halt(0);
247         }
248     }

250     /**
251      * Extra call to ensure exit after long delay
252      */
253     public void launchFinalExit() {
254         Timer timer = new Timer("R66FinalExit", true);
255         ShutdownTimerTask timerTask = new ShutdownTimerTask();
256         timer.schedule(timerTask, shutdownConfiguration.timeout * 4);
257     }

264     private boolean serviceStopped() {
265         if (shutdownConfiguration.serviceFuture != null) {
266             logger.info("Service will be stopped");
267             shutdownConfiguration.serviceFuture.setSuccess();
268             return true;
269         }
270         return false;
271     }

273     private static String applArgs = null;
274     private static volatile boolean shouldRestart = false;
275 
276     /**
277      * Sun property pointing the main class and its arguments.
278      * Might not be defined on non Hotspot VM implementations.
279      */
280     private static final String SUN_JAVA_COMMAND = "sun.java.command";
281 
282     /**
283      * Try to return the application arguments (for Oracle VM)
284      * 
285      * @return null if it cannot
286      */
287     private static String getArgs() {
288         String test = System.getProperty(SUN_JAVA_COMMAND);
289         if (test != null && !test.isEmpty()) {
290             // compute args directly
291             // program main and program arguments
292             String[] mainCommand = test.split(" ");
293             // program main is a jar
294             StringBuilder args = new StringBuilder();
295             if (mainCommand[0].endsWith(".jar")) {
296                 // if it's a jar, add -jar mainJar
297                 args.append("-jar ").append(new File(mainCommand[0]).getPath());
298             } else {
299                 // else it's a .class, add the classpath and mainClass
300                 args.append("-cp \"").append(System.getProperty("java.class.path")).append("\" ")
301                         .append(mainCommand[0]);
302             }
303             // finally add program arguments
304             for (int i = 1; i < mainCommand.length; i++) {
305                 args.append(" ").append(mainCommand[i]);
306             }
307             return args.toString();
308         }
309         return null;
310     }

312     /**
313      * Called to setup main class and args to enable restart
314      * 
315      * @param main
316      * @param args
317      */
318     public static void registerMain(Class<?> main, String[] args) {
319         if (main == null) {
320             applArgs = getArgs();
321             return;
322         }
323         applArgs = ManagementFactory.getRuntimeMXBean().getClassPath();
324         if (applArgs != null && !applArgs.isEmpty()) {
325             applArgs = "-cp " + applArgs;
326         }
327         applArgs += " " + main.getName();
328         for (int i = 0; i < args.length; i++) {
329             applArgs += " " + args[i];
330         }
331     }

333     /**
334      * Restart the application using the preset applArgs and computing the jvmArgs.
335      * execute the command in a shutdown hook, to be sure that all the
336      * resources have been disposed before restarting the application
337      * 
338      * @throws IOException
339      */
340     private static void restartApplication() throws IOException {
341         if (shouldRestart) {
342             try {
343                 // java binary
344                 String java = System.getProperty("java.home") + "/bin/java";
345                 // vm arguments
346                 List<String> vmArguments = ManagementFactory.getRuntimeMXBean().getInputArguments();
347                 StringBuilder vmArgsOneLine = new StringBuilder();
348                 for (String arg : vmArguments) {
349                     // if it's the agent argument : we ignore it otherwise the
350                     // address of the old application and the new one will be in conflict
351                     if (!arg.contains("-agentlib")) {
352                         vmArgsOneLine.append(arg).append(" ");
353                     }
354                 }
355                 // init the command to execute, add the vm args
356                 final StringBuilder cmd;
357                 if (DetectionUtils.isWindows()) {
358                     cmd = new StringBuilder("\"" + java + "\" " + vmArgsOneLine);
359                 } else {
360                     cmd = new StringBuilder(java + " " + vmArgsOneLine);
361                 }
362 
363                 if (applArgs == null) {
364                     applArgs = getArgs();
365                 }
366                 if (applArgs == null) {
367                     // big issue then
368                     System.err.println("Cannot restart!");
369                 }
370                 cmd.append(applArgs);
371                 logger.debug("Should restart with:\n" + cmd.toString());
372                 logger.warn("Should restart");
373                 Runtime.getRuntime().exec(cmd.toString());
374             } catch (Throwable e) {
375                 // something went wrong
376                 throw new IOException("Error while trying to restart the application", e);
377             }
378         }
379     }

398     /**
399      * Intermediary exit function
400      */
401     private static void terminate() {
402         shutdownStarted = true;
403         shutdown = true;
404         if (immediate) {
405             shutdownHook.exit();
406             // Force exit!
407             try {
408                 Thread.sleep(shutdownHook.shutdownConfiguration.timeout / 2);
409             } catch (InterruptedException e) {
410             }
411             if (logger.isDebugEnabled()) {
412                 Map<Thread, StackTraceElement[]> map = Thread
413                         .getAllStackTraces();
414                 for (Thread thread : map.keySet()) {
415                     printStackTrace(thread, map.get(thread));
416                 }
417             }
418             isShutdownOver = true;
419             logger.info("Should restart? " + isRestart());
420             try {
421                 restartApplication();
422             } catch (IOException e1) {
423                 // TODO Auto-generated catch block
424                 e1.printStackTrace();
425             }
426             shutdownHook.serviceStopped();
427             System.err.println("Halt System");
428             try {
429                 Thread.sleep(1000);
430             } catch (InterruptedException e) {
431             }
432             Runtime.getRuntime().halt(0);
433         } else {
434             shutdownHook.launchFinalExit();
435             immediate = true;
436             shutdownHook.exit();
437             isShutdownOver = true;
438             shutdownHook.serviceStopped();
439             logger.info("Should restart? " + isRestart());
440             try {
441                 restartApplication();
442             } catch (IOException e1) {
443                 // TODO Auto-generated catch block
444                 e1.printStackTrace();
445             }
446             logger.info("Exit System");
447             System.err.println("Exit System");
448             //Runtime.getRuntime().halt(0);
449         }
450     }
