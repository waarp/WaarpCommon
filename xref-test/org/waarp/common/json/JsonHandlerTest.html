42  	 * Test method for {@link org.waarp.common.json.JsonHandler#createObjectNode()}.
43  	 */
44  	@Test
45  	public void testCreateObjectNode() {
46  		ObjectNode node = JsonHandler.createObjectNode();
47  		JsonHandler.setValue(node, "boolean", true);
48  		JsonHandler.setValue(node, "bytes", "bytes".getBytes());
49  		JsonHandler.setValue(node, "double", 2.0);
50  		JsonHandler.setValue(node, "int", 3);
51  		JsonHandler.setValue(node, "long", 5L);
52  		JsonHandler.setValue(node, "string", "string");
53  		assertTrue(JsonHandler.exist(node, "boolean"));
54  		assertTrue(JsonHandler.exist(node, "bytes"));
55  		assertTrue(JsonHandler.exist(node, "double"));
56  		assertTrue(JsonHandler.exist(node, "int"));
57  		assertTrue(JsonHandler.exist(node, "long"));
58  		assertTrue(JsonHandler.exist(node, "string"));
59  		assertEquals(true, JsonHandler.getValue(node, "boolean", false));
60  		assertTrue(Arrays.equals("bytes".getBytes(), JsonHandler.getValue(node, "bytes", " ".getBytes())));
61  		assertEquals(2.0, (double)JsonHandler.getValue(node, "double", 1.0), 0.0);
62  		assertEquals(3, (int)JsonHandler.getValue(node, "int", 2));
63  		assertEquals(5L, (long)JsonHandler.getValue(node, "long", 0L));
64  		assertEquals("string", JsonHandler.getValue(node, "string", "no"));
65  		String result = JsonHandler.writeAsString(node);
66  		ObjectNode node2 = JsonHandler.getFromString(result);
67  		assertTrue(result.equals(JsonHandler.writeAsString(node2)));
68  		Map<String, Object> map = JsonHandler.getMapFromString(result);
69  		assertEquals(true, map.get("boolean"));
70  		assertEquals(2.0, (Double)map.get("double"), 0.0);
71  		assertTrue(3 == (Integer)map.get("int"));
72  		assertTrue(5L == (Integer)map.get("long"));
73  		assertEquals("string", map.get("string"));
74  	}

96  		} catch (IOException e) {
97  			fail(e.getMessage());

99  		}
